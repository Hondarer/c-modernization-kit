2.1 callback

PotrThread health_thread;

ヘルスチェックスレッドハンドル (送信者のみ)。

PotrMutex health_mutex;

ヘルスチェックスレッド停止用ミューテックス。

2.5 health_wakeup

PotrCondVar health_wakeup;

ヘルスチェックスレッドを即時起床させる条件変数。

2.6 service

PotrServiceDef service;

サービス定義。

2.7 global

PotrGlobalConfig global;

グローバル設定。

2.8 send_window

PotrWindow send_window;

送信バッファ (過去 N パケット保持。NACK 再送・REJECT 判定に使用)。

2.9 send_window_mutex

PotrMutex send_window_mutex;

send_window 保護用ミューテックス (送信スレッド・ヘルスチェックスレッド・受信スレッドが競合するため)。

2.10 recv_window

PotrWindow recv_window;

受信ウィンドウ (順序整列・欠番検出)。

2.11 sock

PotrSocket[POTR_MAX_PATH] sock;

各パスの UDP ソケット。

2.12 n_path

int n_path;

有効パス数。

2.13 running

volatile int running;

受信スレッド実行フラグ (1: 実行中, 0: 停止)。

2.14 health_running

volatile int health_running;

ヘルスチェックスレッド実行フラグ (1: 実行中, 0: 停止)。

2.15 health_alive

volatile int health_alive;

疎通状態 (1: alive, 0: dead/未接続)。受信者が管理。

2.16 role

PotrRole role;

役割 (POTR_ROLE_SENDER / POTR_ROLE_RECEIVER)。

2.17 src_addr_resolved

struct in_addr[POTR_MAX_PATH] src_addr_resolved;

解決済み送信元 IPv4 アドレス。

2.18 dst_addr_resolved

struct in_addr[POTR_MAX_PATH] dst_addr_resolved;

解決済み宛先 IPv4 アドレス (unicast のみ)。

2.19 dest_addr

struct sockaddr_in[POTR_MAX_PATH] dest_addr;

送信先ソケットアドレス (送信者が sendto に使用)。

uint32_t session_id;

int64_t session_tv_sec;

int32_t session_tv_nsec;

2.24 peer_session_tv_sec

int64_t peer_session_tv_sec;

追跡中の相手セッション開始時刻 秒部。

2.25 peer_session_tv_nsec

int32_t peer_session_tv_nsec;

追跡中の相手セッション開始時刻 ナノ秒部。

2.26 peer_session_known

int peer_session_known;

相手セッションが初期化済みか (0: 未初期化)。

2.27 peer_port

uint16_t[POTR_MAX_PATH] peer_port;

int32_t last_recv_tv_nsec;

2.30 last_recv_tv_sec

int64_t last_recv_tv_sec;

2.32 path_last_recv_nsec

int32_t[POTR_MAX_PATH] path_last_recv_nsec;

パスごとの最終受信時刻 ナノ秒部。

size_t frag_buf_len;

フラグメント結合バッファの現在のデータ長 (バイト)。

2.34 frag_compressed

int frag_compressed;

uint32_t _pad_frag;

size_t compress_buf_size;

compress_buf のサイズ (バイト)。

2.39 crypto_buf

uint8_t *crypto_buf;

2.40 crypto_buf_size

size_t crypto_buf_size;

crypto_buf のサイズ (バイト)。

uint8_t *send_wire_buf;

2.45 _pad_send_thread
