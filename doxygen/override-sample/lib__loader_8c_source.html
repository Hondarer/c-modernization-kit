49void *libbase_load_func_impl(LibFuncCache *cache)
50{
51#ifndef _WIN32
52 const char *ext = ".so";
53#else /* _WIN32 */
54 const char *ext = ".dll";
55#endif /* _WIN32 */
56 char lib_with_ext[LIBBASE_MAX_LIB_NAME];
57
58 /* ロード完了後は func_ptr が NULL 以外になる。
59 * 初回ロード後の読み取りのみになるため早期リターンで判定する。 */
60 if (cache->func_ptr != NULL)
61 {
62 return cache->func_ptr;
63 }
64
65 /* ロード処理を排他制御する。
66 * ロック取得後に再度 func_ptr を確認し、他スレッドが先にロードを
67 * 完了していた場合は処理をスキップする (double-checked locking)。 */
68#ifndef _WIN32
69 if (pthread_mutex_lock(&cache->mutex) != 0)
70 {
71 return NULL;
72 }
73#else /* _WIN32 */
74 AcquireSRWLockExclusive(&cache->lock);
75#endif /* _WIN32 */
76
77 if (cache->func_ptr == NULL)
78 {
79 if (strlen(cache->lib_name) + strlen(ext) >= LIBBASE_MAX_LIB_NAME)
80 {
81 goto unlock;
82 }
83 strcpy(lib_with_ext, cache->lib_name);
84 strcat(lib_with_ext, ext);
85
86#ifndef _WIN32
87 cache->handle = dlopen(lib_with_ext, RTLD_LAZY);
88#else /* _WIN32 */
89 cache->handle = LoadLibrary(lib_with_ext);
90#endif /* _WIN32 */
91 if (cache->handle == NULL)
92 {
93 goto unlock;
94 }
95
96#ifndef _WIN32
97 cache->func_ptr = dlsym(cache->handle, cache->func_name);
98#else /* _WIN32 */
99 cache->func_ptr = (void *)GetProcAddress(cache->handle, cache->func_name);
100#endif /* _WIN32 */
101 if (cache->func_ptr == NULL)
102 {
103#ifndef _WIN32
104 dlclose(cache->handle);
105#else /* _WIN32 */
106 FreeLibrary(cache->handle);
107#endif /* _WIN32 */
108 cache->handle = NULL;
109 }
110 }
111
112unlock:
113#ifndef _WIN32
114 pthread_mutex_unlock(&cache->mutex);
115#else /* _WIN32 */
116 ReleaseSRWLockExclusive(&cache->lock);
117#endif /* _WIN32 */
118
119 return cache->func_ptr;
120}
