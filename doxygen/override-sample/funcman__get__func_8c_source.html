21void *_funcman_get_func(funcman_object *fobj)
22{
23#ifndef _WIN32
24 const char *ext = ".so";
25#else /* _WIN32 */
26 const char *ext = ".dll";
27#endif /* _WIN32 */
28 char lib_with_ext[FUNCMAN_NAME_MAX];
29
30 /* ロード完了後は resolved が 0 以外になる。早期リターンで判定する。
31 * もし解決に失敗している場合は、NULL が返却される。 */
32 if (fobj->resolved != 0)
33 {
34 return fobj->func_ptr;
35 }
36
37 /* ロード処理を排他制御する。
38 * ロック取得後に再度 resolved を確認し、他スレッドが先にロードを
39 * 完了していた場合は処理をスキップする (double-checked locking)。 */
40#ifndef _WIN32
41 if (pthread_mutex_lock(&fobj->mutex) != 0)
42 {
43 return NULL;
44 }
45#else /* _WIN32 */
46 AcquireSRWLockExclusive(&fobj->lock);
47#endif /* _WIN32 */
48
49 if (fobj->resolved == 0)
50 {
51 if (strcmp(fobj->lib_name, "default") == 0 && strcmp(fobj->func_name, "default") == 0)
52 {
53 fobj->resolved = 2; /* resolved=2: 明示的デフォルト */
54 goto unlock;
55 }
56 if (fobj->lib_name[0] == '\0' || fobj->func_name[0] == '\0')
57 {
58 fobj->resolved = -1; /* resolved=-1: 定義なし (定義ファイル不存在、定義行が不存在) */
59 goto unlock;
60 }
61 if (strlen(fobj->lib_name) + strlen(ext) >= FUNCMAN_NAME_MAX)
62 {
63 fobj->resolved = -2; /* resolved=-2: 名称長さオーバー */
64 goto unlock;
65 }
66 strcpy(lib_with_ext, fobj->lib_name);
67 strcat(lib_with_ext, ext);
68
69#ifndef _WIN32
70 fobj->handle = dlopen(lib_with_ext, RTLD_LAZY);
71#else /* _WIN32 */
72 fobj->handle = LoadLibrary(lib_with_ext);
73#endif /* _WIN32 */
74 if (fobj->handle == NULL)
75 {
76 fobj->resolved = -3; /* resolved=-3: ライブラリオープンエラー */
77 goto unlock;
78 }
79
80#ifndef _WIN32
81 fobj->func_ptr = dlsym(fobj->handle, fobj->func_name);
82#else /* _WIN32 */
83 fobj->func_ptr = (void *)GetProcAddress(fobj->handle, fobj->func_name);
84#endif /* _WIN32 */
85 if (fobj->func_ptr == NULL)
86 {
87 fobj->resolved = -4; /* resolved=-4: 関数探索エラー */
88#ifndef _WIN32
89 dlclose(fobj->handle);
90#else /* _WIN32 */
91 FreeLibrary(fobj->handle);
92#endif /* _WIN32 */
93 fobj->handle = NULL;
94 }
95 fobj->resolved = 1; /* resolved=1: 解決済 */
96 }
97
98unlock:
99#ifndef _WIN32
100 pthread_mutex_unlock(&fobj->mutex);
101#else /* _WIN32 */
102 ReleaseSRWLockExclusive(&fobj->lock);
103#endif /* _WIN32 */
104
105 return fobj->func_ptr;
106}
