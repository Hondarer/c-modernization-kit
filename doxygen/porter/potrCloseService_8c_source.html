40/* FIN パケットを全パスへ送信する */
41static void send_fin(struct PotrContext_ *ctx)
42{
43 PotrPacket fin_pkt;
44 PotrPacketSessionHdr shdr;
45 size_t wire_len;
46 int i;
47
48 shdr.service_id = ctx->service.service_id;
49 shdr.session_id = ctx->session_id;
50 shdr.session_tv_sec = ctx->session_tv_sec;
51 shdr.session_tv_nsec = ctx->session_tv_nsec;
52
53 if (packet_build_fin(&fin_pkt, &shdr) != POTR_SUCCESS) return;
54
55 wire_len = packet_wire_size(&fin_pkt);
56
57 for (i = 0; i < ctx->n_path; i++)
58 {
59 if (ctx->sock[i] == POTR_INVALID_SOCKET) continue;
60#ifdef _WIN32
61 sendto(ctx->sock[i], (const char *)&fin_pkt, (int)wire_len, 0,
62 (const struct sockaddr *)&ctx->dest_addr[i],
63 sizeof(ctx->dest_addr[i]));
64#else
65 sendto(ctx->sock[i], &fin_pkt, wire_len, 0,
66 (const struct sockaddr *)&ctx->dest_addr[i],
67 sizeof(ctx->dest_addr[i]));
68#endif
69 }
70}

73POTR_API int POTRAPI potrCloseService(PotrHandle handle)
74{
75 struct PotrContext_ *ctx = (struct PotrContext_ *)handle;
76
77 if (ctx == NULL)
78 {
79 POTR_LOG(POTR_LOG_ERROR, "potrCloseService: handle is NULL");
80 return POTR_ERROR;
81 }
82
83 POTR_LOG(POTR_LOG_INFO,
84 "potrCloseService: service_id=%d closing",
85 ctx->service.service_id);
86
87 /* ヘルスチェックスレッドを停止 (送信者のみ) */
88 if (ctx->health_running)
89 {
90 POTR_LOG(POTR_LOG_DEBUG,
91 "potrCloseService: service_id=%d stopping health thread",
92 ctx->service.service_id);
93 potr_health_thread_stop(ctx);
94 }
95
96 /* 送信スレッドを停止してキューを破棄 (送信者のみ) */
97 if (ctx->send_thread_running)
98 {
99 POTR_LOG(POTR_LOG_DEBUG,
100 "potrCloseService: service_id=%d flushing send queue and sending FIN",
101 ctx->service.service_id);
102 potr_send_queue_wait_drained(&ctx->send_queue);
103 send_fin(ctx);
104 potr_send_thread_stop(ctx);
105 potr_send_queue_destroy(&ctx->send_queue);
106 }
107
108 /* 受信スレッドを停止する */
109 if (ctx->running)
110 {
111 POTR_LOG(POTR_LOG_DEBUG,
112 "potrCloseService: service_id=%d stopping recv thread",
113 ctx->service.service_id);
114 comm_recv_thread_stop(ctx);
115 }
116
117 /* ソケットをクローズ (recv スレッドの有無によらず実施)。
118 Windows: comm_recv_thread_stop 内で closesocket 済みの場合は INVALID_SOCKET になっているため
119 guard により二重 close を回避する。
120 Linux: comm_recv_thread_stop は shutdown のみで close しないため、ここで必ず close する。 */
121 {
122 int i;
123 for (i = 0; i < ctx->n_path; i++)
124 {
125 if (ctx->sock[i] == POTR_INVALID_SOCKET) continue;
126#ifdef _WIN32
127 if (ctx->service.type == POTR_TYPE_MULTICAST)
128 {
129 struct ip_mreq mreq;
130 memset(&mreq, 0, sizeof(mreq));
131 if (parse_ipv4_addr(ctx->service.multicast_group, &mreq.imr_multiaddr)
132 == POTR_SUCCESS)
133 {
134 mreq.imr_interface = ctx->src_addr_resolved[i];
135 setsockopt(ctx->sock[i], IPPROTO_IP, IP_DROP_MEMBERSHIP,
136 (const char *)&mreq, sizeof(mreq));
137 }
138 }
139 closesocket(ctx->sock[i]);
140#else
141 if (ctx->service.type == POTR_TYPE_MULTICAST)
142 {
143 struct ip_mreq mreq;
144 memset(&mreq, 0, sizeof(mreq));
145 if (parse_ipv4_addr(ctx->service.multicast_group, &mreq.imr_multiaddr)
146 == POTR_SUCCESS)
147 {
148 mreq.imr_interface = ctx->src_addr_resolved[i];
149 setsockopt(ctx->sock[i], IPPROTO_IP, IP_DROP_MEMBERSHIP,
150 &mreq, sizeof(mreq));
151 }
152 }
153 close(ctx->sock[i]);
154#endif
155 ctx->sock[i] = POTR_INVALID_SOCKET;
156 }
157 }
158
159#ifdef _WIN32
160 WSACleanup();
161#endif
162
163 /* 送受信ウィンドウと動的バッファを解放 */
164 window_destroy(&ctx->send_window);
165 window_destroy(&ctx->recv_window);
166 free(ctx->frag_buf);
167 free(ctx->compress_buf);
168 free(ctx->crypto_buf);
169 free(ctx->recv_buf);
170 free(ctx->send_wire_buf);
171
172 POTR_LOG(POTR_LOG_INFO,
173 "potrCloseService: service closed");
174
175 free(ctx);
176 return POTR_SUCCESS;
177}
