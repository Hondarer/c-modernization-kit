23POTR_API int POTRAPI potrSend(PotrHandle handle, const void *data, size_t len,
24 int flags)
25{
26 struct PotrContext_ *ctx = (struct PotrContext_ *)handle;
27 const uint8_t *ptr = (const uint8_t *)data;
28 size_t remaining;
29 size_t max_payload;
30 uint16_t base_flags = 0;
31
32 if (ctx == NULL || data == NULL || len == 0
33 || len > (size_t)ctx->global.max_message_size)
34 {
35 POTR_LOG(POTR_LOG_ERROR,
36 "potrSend: invalid argument (handle=%p data=%p len=%zu max=%u)",
37 (const void *)handle, data, len,
38 (ctx != NULL) ? (unsigned)ctx->global.max_message_size : 0U);
39 return POTR_ERROR;
40 }
41
42 POTR_LOG(POTR_LOG_TRACE,
43 "potrSend: service_id=%d len=%zu flags=0x%x",
44 ctx->service.service_id, len, (unsigned)flags);
45
46 /* 圧縮が要求された場合はペイロード全体を圧縮する */
47 if ((flags & POTR_SEND_COMPRESS) != 0)
48 {
49 size_t cmp_len = ctx->compress_buf_size;
50
51 if (potr_compress(ctx->compress_buf, &cmp_len,
52 (const uint8_t *)data, len) != 0)
53 {
54 POTR_LOG(POTR_LOG_ERROR,
55 "potrSend: service_id=%d compression failed (len=%zu)",
56 ctx->service.service_id, len);
57 return POTR_ERROR;
58 }
59
60 if (cmp_len < len)
61 {
62 POTR_LOG(POTR_LOG_TRACE,
63 "potrSend: service_id=%d compress %zu -> %zu bytes",
64 ctx->service.service_id, len, cmp_len);
65 ptr = ctx->compress_buf;
66 len = cmp_len;
67 base_flags = POTR_FLAG_COMPRESSED;
68 }
69 else
70 {
71 POTR_LOG(POTR_LOG_TRACE,
72 "potrSend: service_id=%d compression skipped"
73 " (compressed %zu >= original %zu bytes), sending uncompressed",
74 ctx->service.service_id, cmp_len, len);
75 /* 圧縮効果なし: 非圧縮のまま送信 (ptr, len, base_flags は初期値を維持) */
76 }
77 }
78
79 remaining = len;
80 /* ペイロードエレメントヘッダー (6B) を差し引いた実効フラグメントサイズ上限 */
81 max_payload = ctx->global.max_payload - POTR_PAYLOAD_ELEM_HDR_SIZE;
82 /* 暗号化有効時は GCM 認証タグ (16B) 分をさらに差し引く */
83 if (ctx->service.encrypt_enabled)
84 {
85 max_payload -= POTR_CRYPTO_TAG_SIZE;
86 }
87
88 /* ブロッキング送信: 先行キューの sendto が全完了するまで待機する */
89 if ((flags & POTR_SEND_BLOCKING) != 0)
90 {
91 potr_send_queue_wait_drained(&ctx->send_queue);
92 }
93
94 while (remaining > 0)
95 {
96 size_t chunk;
97 if (remaining > max_payload)
98 {
99 chunk = max_payload;
100 }
101 else
102 {
103 chunk = remaining;
104 }
105 int more_frag = (remaining > chunk);
106 uint16_t elem_flags = base_flags;
107
108 if (more_frag)
109 {
110 elem_flags |= POTR_FLAG_MORE_FRAG;
111 }
112
113 if (potr_send_queue_push_wait(&ctx->send_queue, elem_flags, ptr, (uint16_t)chunk,
114 &ctx->send_thread_running)
115 != POTR_SUCCESS)
116 {
117 POTR_LOG(POTR_LOG_ERROR,
118 "potrSend: service_id=%d send queue push failed"
119 " (send_thread_running=%d)",
120 ctx->service.service_id, ctx->send_thread_running);
121 return POTR_ERROR;
122 }
123
124 ptr += chunk;
125 remaining -= chunk;
126 }
127
128 /* ブロッキング送信: 自身のメッセージの sendto 完了まで待機する */
129 if ((flags & POTR_SEND_BLOCKING) != 0)
130 {
131 potr_send_queue_wait_drained(&ctx->send_queue);
132 }
133
134 return POTR_SUCCESS;
135}
