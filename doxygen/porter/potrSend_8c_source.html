35/* 通信種別に応じた送信先アドレスを構築する */
36static int build_dest_addr(const struct PotrContext_ *ctx,
37 struct sockaddr_in *dest)
38{
39 memset(dest, 0, sizeof(*dest));
40 dest->sin_family = AF_INET;
41
42 switch (ctx->service.type)
43 {
44 case POTR_TYPE_UNICAST:
45 /* unicast: 接続相手のポートへ送信。
46 相手アドレスは受信時に取得するため、ここでは未設定。
47 本実装では送信先ホストは呼び出し元が別途設定すると想定し、
48 サービスの dst_port を宛先ポートとして使用する。 */
49 dest->sin_addr.s_addr = htonl(INADDR_LOOPBACK); /* 暫定: 同一ホスト */
50 dest->sin_port = htons(ctx->service.dst_port);
51 break;
52
53 case POTR_TYPE_MULTICAST:
54 if (parse_ipv4_addr(ctx->service.multicast_group, &dest->sin_addr)
55 != POTR_SUCCESS)
56 {
57 return POTR_ERROR;
58 }
59 dest->sin_port = htons(ctx->service.src_port);
60 break;
61
62 case POTR_TYPE_BROADCAST:
63 if (parse_ipv4_addr(ctx->service.broadcast_addr, &dest->sin_addr)
64 != POTR_SUCCESS)
65 {
66 return POTR_ERROR;
67 }
68 dest->sin_port = htons(ctx->service.src_port);
69 break;
70
71 default:
72 return POTR_ERROR;
73 }
74
75 return POTR_SUCCESS;
76}

79POTR_API int POTRAPI potrSend(PotrHandle handle, const void *data, size_t len,
80 int compress)
81{
82 struct PotrContext_ *ctx = (struct PotrContext_ *)handle;
83 const uint8_t *ptr = (const uint8_t *)data;
84 struct sockaddr_in dest;
85 size_t remaining;
86 size_t max_payload;
87 uint16_t data_flags = POTR_FLAG_DATA;
88
89 if (ctx == NULL || data == NULL || len == 0 || len > POTR_MAX_MESSAGE_SIZE)
90 {
91 return POTR_ERROR;
92 }
93
94 /* 圧縮が要求された場合はペイロード全体を圧縮する */
95 if (compress != 0)
96 {
97 size_t cmp_len = sizeof(ctx->compress_buf);
98
99 if (potr_compress(ctx->compress_buf, &cmp_len,
100 (const uint8_t *)data, len) != 0)
101 {
102 return POTR_ERROR;
103 }
104
105 ptr = ctx->compress_buf;
106 len = cmp_len;
107 data_flags = (uint16_t)(POTR_FLAG_DATA | POTR_FLAG_COMPRESSED);
108 }
109
110 if (build_dest_addr(ctx, &dest) != POTR_SUCCESS)
111 {
112 return POTR_ERROR;
113 }
114
115 remaining = len;
116 max_payload = ctx->global.max_payload;
117
118 while (remaining > 0)
119 {
120 PotrPacket pkt;
121 size_t chunk = (remaining > max_payload) ? max_payload : remaining;
122 int more_frag = (remaining > chunk);
123 uint32_t seq;
124 size_t wire_len;
125
126 /* 送信ウィンドウが満杯の場合はエラー返し（ブロッキングは未実装） */
127 if (window_send_full(&ctx->send_window))
128 {
129 return POTR_ERROR;
130 }
131
132 seq = ctx->send_window.next_seq;
133
134 if (packet_build_data(&pkt, seq, ptr, chunk) != POTR_SUCCESS)
135 {
136 return POTR_ERROR;
137 }
138
139 /* 圧縮フラグと後続フラグメントフラグを付与 */
140 pkt.flags |= (uint16_t)(data_flags & ~(uint16_t)POTR_FLAG_DATA);
141 if (more_frag)
142 {
143 pkt.flags |= POTR_FLAG_MORE_FRAG;
144 }
145
146 if (window_send_push(&ctx->send_window, &pkt) != POTR_SUCCESS)
147 {
148 return POTR_ERROR;
149 }
150
151 wire_len = packet_wire_size(&pkt);
152
153#ifdef _WIN32
154 if (sendto(ctx->sock, (const char *)&pkt, (int)wire_len, 0,
155 (const struct sockaddr *)&dest, sizeof(dest)) == SOCKET_ERROR)
156 {
157 return POTR_ERROR;
158 }
159#else
160 if (sendto(ctx->sock, &pkt, wire_len, 0,
161 (const struct sockaddr *)&dest, sizeof(dest)) < 0)
162 {
163 return POTR_ERROR;
164 }
165#endif
166
167 ptr += chunk;
168 remaining -= chunk;
169 }
170
171 return POTR_SUCCESS;
172}
