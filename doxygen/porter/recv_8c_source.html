75static void on_recv(int service_id, PotrEvent event,
76 const void *data, size_t len)
77{
78 char buf[POTR_MAX_PAYLOAD + 1];
79 size_t copy_len;
80
81 switch (event)
82 {
83 case POTR_EVENT_CONNECTED:
84 printf("[サービス %d] 接続確立\n", service_id);
85 break;
86
87 case POTR_EVENT_DISCONNECTED:
88 printf("[サービス %d] 切断検知\n", service_id);
89 break;
90
91 case POTR_EVENT_DATA:
92 default:
93 if (len < POTR_MAX_PAYLOAD)
94 {
95 copy_len = len;
96 }
97 else
98 {
99 copy_len = POTR_MAX_PAYLOAD;
100 }
101 memcpy(buf, data, copy_len);
102 buf[copy_len] = '\0';
103 printf("[サービス %d] 受信 (%zu バイト): %s\n", service_id, len, buf);
104 break;
105 }
106}

116static int parse_log_level(const char *str, PotrLogLevel *out)
117{
118 static const struct { const char *name; PotrLogLevel level; uint32_t _pad; } tbl[] = {
119 { "TRACE", POTR_LOG_TRACE, 0U },
120 { "DEBUG", POTR_LOG_DEBUG, 0U },
121 { "INFO", POTR_LOG_INFO, 0U },
122 { "WARN", POTR_LOG_WARN, 0U },
123 { "ERROR", POTR_LOG_ERROR, 0U },
124 { "FATAL", POTR_LOG_FATAL, 0U },
125 };
126 char upper[16];
127 size_t i;
128 size_t j;
129
130 for (j = 0; j < sizeof(upper) - 1U && str[j] != '\0'; j++)
131 {
132 upper[j] = (str[j] >= 'a' && str[j] <= 'z')
133 ? (char)(str[j] - ('a' - 'A'))
134 : str[j];
135 }
136 upper[j] = '\0';
137
138 for (i = 0; i < sizeof(tbl) / sizeof(tbl[0]); i++)
139 {
140 if (strcmp(upper, tbl[i].name) == 0)
141 {
142 *out = tbl[i].level;
143 return 1;
144 }
145 }
146 return 0;
147}
