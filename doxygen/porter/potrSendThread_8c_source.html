57/* 現在時刻をミリ秒単位で返す (単調増加クロック) */
58static uint64_t get_ms(void)
59{
60#ifdef _WIN32
61 return (uint64_t)GetTickCount64();
62#else
63 struct timespec ts;
64 clock_gettime(CLOCK_MONOTONIC, &ts);
65 return (uint64_t)ts.tv_sec * 1000ULL + (uint64_t)ts.tv_nsec / 1000000ULL;
66#endif
67}

69/* ペイロードエレメントを packed_buf に追記する */
70static void append_payload_elem(uint8_t *packed_buf, size_t *packed_len,
71 const PotrPayloadElem *entry)
72{
73 uint16_t flags_nbo = htons(entry->flags);
74 uint32_t plen_nbo = htonl((uint32_t)entry->payload_len);
75
76 memcpy(packed_buf + *packed_len, &flags_nbo, 2); *packed_len += 2;
77 memcpy(packed_buf + *packed_len, &plen_nbo, 4); *packed_len += 4;
78 memcpy(packed_buf + *packed_len, entry->payload, entry->payload_len);
79 *packed_len += entry->payload_len;
80}

82/* send_wire_buf の [PACKET_HEADER_SIZE .. PACKET_HEADER_SIZE+packed_len-1] に
83 詰め済みのペイロードから外側コンテナを構築して sendto する。
84 seq_num を付与し、再送バッファ (send_window) にも登録する。
85 send_wire_buf = [NBO ヘッダー 32B][packed_payload packed_len B] として組み立てる。 */
86static void flush_packed(struct PotrContext_ *ctx, size_t packed_len)
87{
88 PotrPacket outer_pkt;
89 PotrPacketSessionHdr shdr;
90 uint32_t seq;
91 size_t wire_len;
92 uint8_t *packed_buf = ctx->send_wire_buf + PACKET_HEADER_SIZE;
93
94 shdr.service_id = ctx->service.service_id;
95 shdr.session_id = ctx->session_id;
96 shdr.session_tv_sec = ctx->session_tv_sec;
97 shdr.session_tv_nsec = ctx->session_tv_nsec;
98 shdr._pad = 0;
99
100 /* send_window へのアクセスを排他制御する (送信スレッド・ヘルスチェックスレッド・受信スレッドが競合) */
101#ifdef _WIN32
102 EnterCriticalSection(&ctx->send_window_mutex);
103#else
104 pthread_mutex_lock(&ctx->send_window_mutex);
105#endif
106
107 seq = ctx->send_window.next_seq;
108
109 if (packet_build_packed(&outer_pkt, &shdr, seq, packed_buf, packed_len)
110 != POTR_SUCCESS)
111 {
112#ifdef _WIN32
113 LeaveCriticalSection(&ctx->send_window_mutex);
114#else
115 pthread_mutex_unlock(&ctx->send_window_mutex);
116#endif
117 return;
118 }
119
120 if (ctx->service.encrypt_enabled)
121 {
122 /* 暗号化パス:
123 * 1. ENCRYPTED フラグを OR (outer_pkt.flags は既に NBO)
124 * 2. payload_len を packed_len + TAG_SIZE に更新
125 * 3. nonce = session_id(NBO,4B) + seq_num(NBO,4B) + 0x00*4
126 * 4. AAD = outer_pkt ヘッダー 32B (NBO)
127 * 5. packed_buf → ctx->crypto_buf に暗号化
128 * 6. window_send_push には暗号化済みペイロードを登録
129 * 7. send_wire_buf に暗号化済みデータを組立て
130 */
131 uint8_t nonce[POTR_CRYPTO_NONCE_SIZE];
132 size_t enc_len = ctx->crypto_buf_size;
133
134 outer_pkt.flags |= htons(POTR_FLAG_ENCRYPTED);
135 outer_pkt.payload_len = htons((uint16_t)(packed_len + POTR_CRYPTO_TAG_SIZE));
136
137 /* ノンス: session_id と seq_num は outer_pkt では NBO */
138 memcpy(nonce, &outer_pkt.session_id, 4);
139 memcpy(nonce + 4, &outer_pkt.seq_num, 4);
140 memset(nonce + 8, 0, 4);
141
142 if (potr_encrypt(ctx->crypto_buf, &enc_len,
143 packed_buf, packed_len,
144 ctx->service.encrypt_key,
145 nonce,
146 (const uint8_t *)&outer_pkt, PACKET_HEADER_SIZE) != 0)
147 {
148#ifdef _WIN32
149 LeaveCriticalSection(&ctx->send_window_mutex);
150#else
151 pthread_mutex_unlock(&ctx->send_window_mutex);
152#endif
153 POTR_LOG(POTR_LOG_ERROR,
154 "sender[service_id=%d]: encrypt failed seq=%u",
155 ctx->service.service_id, (unsigned)seq);
156 return;
157 }
158
159 /* window には暗号化済みペイロードを格納して NACK 再送時に再暗号化不要にする */
160 outer_pkt.payload = ctx->crypto_buf;
161 window_send_push(&ctx->send_window, &outer_pkt);
162
163#ifdef _WIN32
164 LeaveCriticalSection(&ctx->send_window_mutex);
165#else
166 pthread_mutex_unlock(&ctx->send_window_mutex);
167#endif
168
169 /* wire 組立: NBO ヘッダー + 暗号文 + タグ */
170 memcpy(ctx->send_wire_buf, &outer_pkt, PACKET_HEADER_SIZE);
171 memcpy(ctx->send_wire_buf + PACKET_HEADER_SIZE, ctx->crypto_buf, enc_len);
172 wire_len = PACKET_HEADER_SIZE + enc_len;
173
174 POTR_LOG(POTR_LOG_TRACE,
175 "sender[service_id=%d]: DATA(enc) seq=%u packed_len=%zu enc_len=%zu",
176 ctx->service.service_id, (unsigned)seq, packed_len, enc_len);
177 }
178 else
179 {
180 window_send_push(&ctx->send_window, &outer_pkt);
181
182#ifdef _WIN32
183 LeaveCriticalSection(&ctx->send_window_mutex);
184#else
185 pthread_mutex_unlock(&ctx->send_window_mutex);
186#endif
187
188 /* NBO ヘッダー (32B) を send_wire_buf 先頭に書き込む (ペイロードは既に直後に配置済み) */
189 memcpy(ctx->send_wire_buf, &outer_pkt, PACKET_HEADER_SIZE);
190 wire_len = PACKET_HEADER_SIZE + packed_len;
191
192 POTR_LOG(POTR_LOG_TRACE,
193 "sender[service_id=%d]: DATA seq=%u packed_len=%zu",
194 ctx->service.service_id, (unsigned)seq, packed_len);
195 }
196
197 {
198 int i;
199 for (i = 0; i < ctx->n_path; i++)
200 {
201#ifdef _WIN32
202 sendto(ctx->sock[i], (const char *)ctx->send_wire_buf, (int)wire_len, 0,
203 (const struct sockaddr *)&ctx->dest_addr[i],
204 sizeof(ctx->dest_addr[i]));
205#else
206 sendto(ctx->sock[i], ctx->send_wire_buf, wire_len, 0,
207 (const struct sockaddr *)&ctx->dest_addr[i],
208 sizeof(ctx->dest_addr[i]));
209#endif
210 }
211 }
212
213 /* ヘルスチェックスレッドが参照する最終送信時刻を更新し、
214 スリープ中のヘルスチェックスレッドを起床させてタイマーをリセットする */
215 ctx->last_send_ms = get_ms();
216
217 if (ctx->health_running)
218 {
219#ifdef _WIN32
220 EnterCriticalSection(&ctx->health_mutex);
221 WakeConditionVariable(&ctx->health_wakeup);
222 LeaveCriticalSection(&ctx->health_mutex);
223#else
224 pthread_mutex_lock(&ctx->health_mutex);
225 pthread_cond_signal(&ctx->health_wakeup);
226 pthread_mutex_unlock(&ctx->health_mutex);
227#endif
228 }
229}

231/* 送信スレッド本体 */
232#ifdef _WIN32
233static DWORD WINAPI send_thread_func(LPVOID arg)
234#else
235static void *send_thread_func(void *arg)
236#endif
237{
238 struct PotrContext_ *ctx = (struct PotrContext_ *)arg;
239 PotrPayloadElem first;
240
241 for (;;)
242 {
243 /* キューからエントリを取り出す (ブロッキング) */
244 if (potr_send_queue_pop(&ctx->send_queue, &first,
245 &ctx->send_thread_running) != POTR_SUCCESS)
246 {
247 break;
248 }
249
250 /* パッキング試行 */
251 {
252 /* packed_buf は send_wire_buf のヘッダー直後領域を直接使用 (ゼロコピー) */
253 uint8_t *packed_buf = ctx->send_wire_buf + PACKET_HEADER_SIZE;
254 size_t packed_len = 0;
255 int n_dequeued = 1;
256
257 append_payload_elem(packed_buf, &packed_len, &first);
258
259 /* MORE_FRAG エントリはパッキング不可。そのまま単体コンテナとして送信。 */
260 if (!(first.flags & POTR_FLAG_MORE_FRAG))
261 {
262 uint32_t pack_wait_ms = ctx->service.pack_wait_ms;
263
264 if (pack_wait_ms > 0)
265 {
266 /* パッキング待ちあり: タイムアウトまで追加エントリを待ち合わせる */
267 uint64_t deadline = get_ms() + pack_wait_ms;
268 PotrPayloadElem next;
269
270 for (;;)
271 {
272 uint64_t now = get_ms();
273 uint32_t remaining;
274 size_t elem_size;
275
276 if (now >= deadline)
277 {
278 break; /* タイムアウト */
279 }
280
281 remaining = (uint32_t)(deadline - now);
282
283 if (potr_send_queue_peek_timed(&ctx->send_queue, &next, remaining)
284 != POTR_SUCCESS)
285 {
286 break; /* タイムアウト (エントリなし) */
287 }
288
289 if (next.flags & POTR_FLAG_MORE_FRAG)
290 {
291 break; /* MORE_FRAG はパッキング不可 */
292 }
293
294 elem_size = POTR_PAYLOAD_ELEM_HDR_SIZE + (size_t)next.payload_len;
295
296 if (packed_len + elem_size > (size_t)ctx->global.max_payload
297 - (ctx->service.encrypt_enabled
298 ? POTR_CRYPTO_TAG_SIZE : 0U))
299 {
300 break; /* 容量満杯: 即時送信してタイマーリセット */
301 }
302
303 if (potr_send_queue_try_pop(&ctx->send_queue, &next)
304 != POTR_SUCCESS)
305 {
306 break; /* 競合防止 (通常発生しない) */
307 }
308
309 append_payload_elem(packed_buf, &packed_len, &next);
310 n_dequeued++;
311 }
312 }
313 else
314 {
315 /* パッキング待ちなし: キューにあるエントリを即時まとめる */
316 PotrPayloadElem next;
317
318 while (potr_send_queue_peek(&ctx->send_queue, &next) == POTR_SUCCESS)
319 {
320 size_t elem_size;
321
322 if (next.flags & POTR_FLAG_MORE_FRAG)
323 {
324 break;
325 }
326
327 elem_size = POTR_PAYLOAD_ELEM_HDR_SIZE + (size_t)next.payload_len;
328
329 if (packed_len + elem_size > (size_t)ctx->global.max_payload
330 - (ctx->service.encrypt_enabled
331 ? POTR_CRYPTO_TAG_SIZE : 0U))
332 {
333 break;
334 }
335
336 if (potr_send_queue_try_pop(&ctx->send_queue, &next)
337 != POTR_SUCCESS)
338 {
339 break;
340 }
341
342 append_payload_elem(packed_buf, &packed_len, &next);
343 n_dequeued++;
344 }
345 }
346 }
347
348 /* 外側パケットを構築して送信 */
349 flush_packed(ctx, packed_len);
350
351 /* デキューした全エントリ分の inflight を減算 */
352 {
353 int i;
354 for (i = 0; i < n_dequeued; i++)
355 {
356 potr_send_queue_complete(&ctx->send_queue);
357 }
358 }
359 }
360 }
361
362#ifdef _WIN32
363 return 0;
364#else
365 return NULL;
366#endif
367}

369/* doxygen コメントは、ヘッダに記載 */
370int potr_send_thread_start(struct PotrContext_ *ctx)
371{
372 ctx->send_thread_running = 1;
373
374#ifdef _WIN32
375 InitializeCriticalSection(&ctx->send_window_mutex);
376 ctx->send_thread = CreateThread(NULL, 0, send_thread_func, ctx, 0, NULL);
377 if (ctx->send_thread == NULL)
378 {
379 ctx->send_thread_running = 0;
380 DeleteCriticalSection(&ctx->send_window_mutex);
381 return POTR_ERROR;
382 }
383#else
384 pthread_mutex_init(&ctx->send_window_mutex, NULL);
385 if (pthread_create(&ctx->send_thread, NULL, send_thread_func, ctx) != 0)
386 {
387 ctx->send_thread_running = 0;
388 pthread_mutex_destroy(&ctx->send_window_mutex);
389 return POTR_ERROR;
390 }
391#endif
392
393 return POTR_SUCCESS;
394}

397void potr_send_thread_stop(struct PotrContext_ *ctx)
398{
399 ctx->send_thread_running = 0;
400 potr_send_queue_shutdown(&ctx->send_queue);
401
402#ifdef _WIN32
403 WaitForSingleObject(ctx->send_thread, INFINITE);
404 CloseHandle(ctx->send_thread);
405 DeleteCriticalSection(&ctx->send_window_mutex);
406#else

408 pthread_mutex_destroy(&ctx->send_window_mutex);
409#endif
410}
