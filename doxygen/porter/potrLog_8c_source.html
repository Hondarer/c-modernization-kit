125static const char *log_basename(const char *path)
126{
127 const char *p = path;
128 const char *last = path;
129
130 while (*p != '\0')
131 {
132 if (*p == '/' || *p == '\\')
133 {
134 last = p + 1;
135 }
136 p++;
137 }
138 return last;
139}

146static void log_timestamp(char *buf, size_t buflen)
147{
148#ifdef _WIN32
149 SYSTEMTIME st;
150 GetLocalTime(&st);
151 /* snprintf は _snprintf_s でなく標準の snprintf を使用 (C99 準拠)。 */
152 (void)snprintf(buf, buflen,
153 "[%04d-%02d-%02d %02d:%02d:%02d.%03d]",
154 (int)st.wYear, (int)st.wMonth, (int)st.wDay,
155 (int)st.wHour, (int)st.wMinute, (int)st.wSecond,
156 (int)st.wMilliseconds);
157#else
158 struct timespec ts;
159 struct tm tm_info;
160
161 clock_gettime(CLOCK_REALTIME, &ts);
162 localtime_r(&ts.tv_sec, &tm_info);
163 (void)snprintf(buf, buflen,
164 "[%04d-%02d-%02d %02d:%02d:%02d.%03d]",
165 tm_info.tm_year + 1900, tm_info.tm_mon + 1, tm_info.tm_mday,
166 tm_info.tm_hour, tm_info.tm_min, tm_info.tm_sec,
167 (int)(ts.tv_nsec / 1000000L));
168#endif
169}

194POTR_API int POTRAPI potrLogConfig(PotrLogLevel level,
195 const char *log_file,
196 int console)
197{
198 FILE *new_fp = NULL;
199
200 /* log_file を事前にオープンしておく (ミューテックス外で実施し、ロック時間を短縮)。 */
201 if (log_file != NULL && log_file[0] != '\0')
202 {
203#ifdef _WIN32
204 if (fopen_s(&new_fp, log_file, "a") != 0)
205 {
206 new_fp = NULL;
207 }
208#else
209 new_fp = fopen(log_file, "a");
210#endif
211 if (new_fp == NULL)
212 {
213 return POTR_ERROR;
214 }
215 }
216
217 log_lock();
218
219 /* 既存ファイルをクローズ。 */
220 if (g_log_fp != NULL)
221 {
222 fclose(g_log_fp);
223 g_log_fp = NULL;
224 }
225
226#ifndef _WIN32
227 /* 既存 syslog をクローズして再度 openlog する (ident は "porter" 固定)。 */
228 if (g_syslog_open)
229 {
230 closelog();
231 g_syslog_open = 0;
232 }
233#endif
234
235 /* 設定を更新する。 */
236 g_log_level = (int)level;
237 g_log_console = console;
238 g_log_fp = new_fp;
239
240 if (log_file != NULL && log_file[0] != '\0')
241 {
242 size_t len = strlen(log_file);
243 if (len >= POTR_LOG_FILE_PATH_MAX)
244 {
245 len = POTR_LOG_FILE_PATH_MAX - 1;
246 }
247 memcpy(g_log_file_path, log_file, len);
248 g_log_file_path[len] = '\0';
249 }
250 else
251 {
252 g_log_file_path[0] = '\0';
253 }
254
255#ifndef _WIN32
256 /* level が POTR_LOG_OFF でない場合のみ syslog を開く。 */
257 if (level != POTR_LOG_OFF)
258 {
259 openlog("porter", LOG_PID | LOG_NDELAY, LOG_USER);
260 g_syslog_open = 1;
261 }
262#endif
263
264 log_unlock();
265
266 return POTR_SUCCESS;
267}

270void potr_log_write(PotrLogLevel level, const char *file, int line,
271 const char *fmt, ...)
272{
273 char msg[512];
274 char ts[96];
275 va_list ap;
276 int cur_level;
277
278 /* クイックチェック (ロックなし)。ほとんどの場合ここで早期リターンする。 */
279 cur_level = g_log_level;
280 if ((int)level < cur_level || cur_level >= (int)POTR_LOG_OFF)
281 {
282 return;
283 }
284
285 /* メッセージを先にフォーマット (ロック外でスループット向上)。 */
286 va_start(ap, fmt);
287 (void)vsnprintf(msg, sizeof(msg), fmt, ap);
288 va_end(ap);
289
290 log_lock();
291
292 /* ロック後に再確認 (設定変更の競合を回避)。 */
293 cur_level = g_log_level;
294 if ((int)level < cur_level || cur_level >= (int)POTR_LOG_OFF)
295 {
296 log_unlock();
297 return;
298 }
299
300 {
301 const char *lstr = ((int)level >= 0 && level <= POTR_LOG_FATAL)
302 ? s_level_str[(int)level]
303 : "?????";
304
305#ifndef _WIN32
306 /* ── Linux: syslog ─────────────────────────────────────────────── */
307 if (g_syslog_open)
308 {
309 syslog(level_to_syslog_priority(level),
310 "[%s] [%s:%d] %s",
311 lstr, log_basename(file), line, msg);
312 }
313#endif
314
315 /* ── ファイル / stderr: タイムスタンプ付きフォーマット ─────────── */
316 if (g_log_fp != NULL || g_log_console)
317 {
318 log_timestamp(ts, sizeof(ts));
319
320 if (g_log_fp != NULL)
321 {
322 (void)fprintf(g_log_fp, "%s [%s] [%s:%d] %s\n",
323 ts, lstr, log_basename(file), line, msg);
324 (void)fflush(g_log_fp);
325 }
326
327 if (g_log_console)
328 {
329 (void)fprintf(stderr, "%s [%s] [%s:%d] %s\n",
330 ts, lstr, log_basename(file), line, msg);
331 }
332 }
333 }
334
335 log_unlock();
336}
