51int potr_send_queue_init(PotrSendQueue *q, size_t depth, uint16_t max_payload)
52{
53 size_t i;
54
55 memset(q, 0, sizeof(*q));
56
57 q->entries = (PotrPayloadElem *)malloc(depth * sizeof(PotrPayloadElem));
58 q->payload_pool = (uint8_t *)malloc(depth * (size_t)max_payload);
59
60 if (q->entries == NULL || q->payload_pool == NULL)
61 {
62 free(q->entries);
63 free(q->payload_pool);
64 q->entries = NULL;
65 q->payload_pool = NULL;
66 return POTR_ERROR;
67 }
68
69 q->depth = depth;
70
71 for (i = 0; i < depth; i++)
72 {
73 q->entries[i].flags = 0;
74 q->entries[i].payload_len = 0;
75 q->entries[i].payload = q->payload_pool + i * (size_t)max_payload;
76 }
77
78 POTR_MUTEX_INIT(&q->mutex);
79 POTR_COND_INIT(&q->not_empty);
80 POTR_COND_INIT(&q->not_full);
81 POTR_COND_INIT(&q->drained);
82 return POTR_SUCCESS;
83}

99int potr_send_queue_push(PotrSendQueue *q, uint16_t flags,
100 const void *payload, uint16_t payload_len)
101{
102 POTR_MUTEX_LOCK(&q->mutex);
103
104 if (q->count + q->inflight >= q->depth)
105 {
106 POTR_MUTEX_UNLOCK(&q->mutex);
107 return POTR_ERROR;
108 }
109
110 q->entries[q->tail].flags = flags;
111 q->entries[q->tail].payload_len = payload_len;
112 memcpy(q->entries[q->tail].payload, payload, payload_len);
113 q->tail = (q->tail + 1U) % q->depth;
114 q->count++;
115
116 POTR_COND_SIGNAL(&q->not_empty);
117 POTR_MUTEX_UNLOCK(&q->mutex);
118
119 return POTR_SUCCESS;
120}

123int potr_send_queue_push_wait(PotrSendQueue *q, uint16_t flags,
124 const void *payload, uint16_t payload_len,
125 volatile int *running)
126{
127 POTR_MUTEX_LOCK(&q->mutex);
128
129 /* count + inflight < depth が保証されるまで待機する。
130 inflight エントリもプールスロットを占有するため、count だけでは不足。 */
131 while (q->count + q->inflight >= q->depth)
132 {
133 if (!*running)
134 {
135 POTR_MUTEX_UNLOCK(&q->mutex);
136 return POTR_ERROR;
137 }
138 POTR_COND_WAIT(&q->not_full, &q->mutex);
139 }
140
141 q->entries[q->tail].flags = flags;
142 q->entries[q->tail].payload_len = payload_len;
143 memcpy(q->entries[q->tail].payload, payload, payload_len);
144 q->tail = (q->tail + 1U) % q->depth;
145 q->count++;
146
147 POTR_COND_SIGNAL(&q->not_empty);
148 POTR_MUTEX_UNLOCK(&q->mutex);
149
150 return POTR_SUCCESS;
151}

154int potr_send_queue_pop(PotrSendQueue *q, PotrPayloadElem *out, volatile int *running)
155{
156 POTR_MUTEX_LOCK(&q->mutex);
157
158 while (q->count == 0)
159 {
160 if (!*running)
161 {
162 POTR_MUTEX_UNLOCK(&q->mutex);
163 return POTR_ERROR;
164 }
165 POTR_COND_WAIT(&q->not_empty, &q->mutex);
166 }
167
168 *out = q->entries[q->head];
169 q->head = (q->head + 1U) % q->depth;
170 q->count--;
171 q->inflight++;
172
173 /* count + inflight は変化しない (count-- と inflight++ が相殺) ため
174 not_full シグナルは complete() が担う */
175 POTR_MUTEX_UNLOCK(&q->mutex);
176 return POTR_SUCCESS;
177}

197int potr_send_queue_peek_timed(PotrSendQueue *q, PotrPayloadElem *out,
198 uint32_t timeout_ms)
199{
200 POTR_MUTEX_LOCK(&q->mutex);
201
202 if (q->count == 0)
203 {
204#ifdef _WIN32
205 SleepConditionVariableCS(&q->not_empty, &q->mutex, (DWORD)timeout_ms);
206#else
207 struct timespec abs_ts;
208 clock_gettime(CLOCK_REALTIME, &abs_ts);
209 abs_ts.tv_sec += (time_t)(timeout_ms / 1000U);
210 abs_ts.tv_nsec += (long)((timeout_ms % 1000U) * 1000000UL);
211 if (abs_ts.tv_nsec >= 1000000000L)
212 {
213 abs_ts.tv_sec++;
214 abs_ts.tv_nsec -= 1000000000L;
215 }
216 pthread_cond_timedwait(&q->not_empty, &q->mutex, &abs_ts);
217#endif
218 }
219
220 if (q->count == 0)
221 {
222 POTR_MUTEX_UNLOCK(&q->mutex);
223 return POTR_ERROR;
224 }
225
226 *out = q->entries[q->head];
227
228 POTR_MUTEX_UNLOCK(&q->mutex);
229 return POTR_SUCCESS;
230}

252/* doxygen コメントは、ヘッダに記載 */
253void potr_send_queue_complete(PotrSendQueue *q)
254{
255 POTR_MUTEX_LOCK(&q->mutex);
256
257 if (q->inflight > 0U)
258 {
259 q->inflight--;
260 }
261
262 if (q->count == 0U && q->inflight == 0U)
263 {
264 POTR_COND_BROADCAST(&q->drained);
265 }
266
267 /* inflight 減少により count + inflight < depth となる可能性があるため
268 push_wait で待機中のスレッドを起床させる */
269 POTR_COND_SIGNAL(&q->not_full);
270
271 POTR_MUTEX_UNLOCK(&q->mutex);
272}
