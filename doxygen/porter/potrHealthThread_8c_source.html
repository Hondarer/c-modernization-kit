40/* health_interval_ms ミリ秒、または停止シグナルが来るまでスリープする */
41static void health_sleep(struct PotrContext_ *ctx, uint32_t interval_ms)
42{
43#ifdef _WIN32
44 EnterCriticalSection(&ctx->health_mutex);
45 if (ctx->health_running)
46 {
47 SleepConditionVariableCS(&ctx->health_wakeup, &ctx->health_mutex, (DWORD)interval_ms);
48 }
49 LeaveCriticalSection(&ctx->health_mutex);
50#else
51 struct timespec abs_ts;
52 clock_gettime(CLOCK_REALTIME, &abs_ts);
53 abs_ts.tv_sec += (time_t)(interval_ms / 1000U);
54 abs_ts.tv_nsec += (long)((interval_ms % 1000U) * 1000000UL);
55 if (abs_ts.tv_nsec >= 1000000000L)
56 {
57 abs_ts.tv_sec++;
58 abs_ts.tv_nsec -= 1000000000L;
59 }
60 pthread_mutex_lock(&ctx->health_mutex);
61 if (ctx->health_running)
62 {
63 pthread_cond_timedwait(&ctx->health_wakeup, &ctx->health_mutex, &abs_ts);
64 }
65 pthread_mutex_unlock(&ctx->health_mutex);
66#endif
67}

69/* 現在時刻をミリ秒単位で返す (単調増加クロック) */
70static uint64_t health_get_ms(void)
71{
72#ifdef _WIN32
73 return (uint64_t)GetTickCount64();
74#else
75 struct timespec ts;
76 clock_gettime(CLOCK_MONOTONIC, &ts);
77 return (uint64_t)ts.tv_sec * 1000ULL + (uint64_t)ts.tv_nsec / 1000000ULL;
78#endif
79}

81/* ヘルスチェックスレッド本体 */
82#ifdef _WIN32
83static DWORD WINAPI health_thread_func(LPVOID arg)
84#else
85static void *health_thread_func(void *arg)
86#endif
87{
88 struct PotrContext_ *ctx = (struct PotrContext_ *)arg;
89 PotrPacketSessionHdr shdr;
90
91 shdr.service_id = ctx->service.service_id;
92 shdr.session_id = ctx->session_id;
93 shdr.session_tv_sec = ctx->session_tv_sec;
94 shdr.session_tv_nsec = ctx->session_tv_nsec;
95 shdr._pad = 0;
96
97 while (ctx->health_running)
98 {
99 /* 最終送信 (データ or PING) から health_interval_ms 経過するまで待機する。
100 スリープ時間を動的に計算することで、直後にデータ送信があっても
101 health_interval_ms 周期が崩れないようにする。 */
102 {
103 uint64_t last_send = ctx->last_send_ms; /* volatile 読み取り */
104 uint32_t sleep_ms;
105
106 if (last_send == 0)
107 {
108 /* 未送信: 通常の間隔でそのままスリープ */
109 sleep_ms = ctx->global.health_interval_ms;
110 }
111 else
112 {
113 uint64_t now = health_get_ms();
114 uint64_t elapsed = now - last_send;
115
116 if (elapsed >= (uint64_t)ctx->global.health_interval_ms)
117 {
118 /* 既に間隔を超過: 即時送信するため極短時間だけ待つ */
119 sleep_ms = 1U;
120 }
121 else
122 {
123 /* 残り時間だけスリープ */
124 sleep_ms = (uint32_t)(ctx->global.health_interval_ms - elapsed);
125 }
126 }
127
128 health_sleep(ctx, sleep_ms);
129 }
130
131 if (!ctx->health_running)
132 {
133 break;
134 }
135
136 /* データ送信によりタイマーがリセットされた場合は PING 不要 */
137 {
138 uint64_t last_send = ctx->last_send_ms; /* volatile 読み取り */
139 if (last_send != 0)
140 {
141 uint64_t now = health_get_ms();
142 uint64_t elapsed = now - last_send;
143
144 if (elapsed < (uint64_t)ctx->global.health_interval_ms)
145 {
146 POTR_LOG(POTR_LOG_TRACE,
147 "health[service_id=%d]: PING timer reset",
148 ctx->service.service_id);
149 continue;
150 }
151 }
152 }
153
154 /* PING を送信する (ウィンドウには登録しない: NACK・再送の対象外) */
155 {
156 PotrPacket ping_pkt;
157 uint32_t seq;
158 size_t wire_len;
159 int build_result;
160
161 /* send_window へのアクセスを排他制御する (next_seq の読み取りにのみ使用) */
162#ifdef _WIN32
163 EnterCriticalSection(&ctx->send_window_mutex);
164#else
165 pthread_mutex_lock(&ctx->send_window_mutex);
166#endif
167
168 /* next_seq を読み取る (PING は消費しない: window_send_push は呼ばない) */
169 seq = ctx->send_window.next_seq;
170 build_result = packet_build_ping(&ping_pkt, &shdr, seq);
171
172#ifdef _WIN32
173 LeaveCriticalSection(&ctx->send_window_mutex);
174#else
175 pthread_mutex_unlock(&ctx->send_window_mutex);
176#endif
177
178 if (build_result != POTR_SUCCESS)
179 {
180 continue;
181 }
182
183 POTR_LOG(POTR_LOG_TRACE,
184 "health[service_id=%d]: PING seq=%u",
185 ctx->service.service_id, (unsigned)seq);
186
187 wire_len = packet_wire_size(&ping_pkt);
188
189 {
190 int k;
191 for (k = 0; k < ctx->n_path; k++)
192 {
193#ifdef _WIN32
194 sendto(ctx->sock[k], (const char *)&ping_pkt, (int)wire_len, 0,
195 (const struct sockaddr *)&ctx->dest_addr[k],
196 sizeof(ctx->dest_addr[k]));
197#else
198 sendto(ctx->sock[k], &ping_pkt, wire_len, 0,
199 (const struct sockaddr *)&ctx->dest_addr[k],
200 sizeof(ctx->dest_addr[k]));
201#endif
202 }
203 }
204
205 /* 最終送信時刻を更新する (次回の待機時間計算に使用) */
206 ctx->last_send_ms = health_get_ms();
207 }
208 }
209
210#ifdef _WIN32
211 return 0;
212#else
213 return NULL;
214#endif

227int potr_health_thread_start(struct PotrContext_ *ctx)
228{
229 if (ctx == NULL)
230 {
231 return POTR_ERROR;
232 }
233
234 if (ctx->global.health_interval_ms == 0)
235 {
236 POTR_LOG(POTR_LOG_DEBUG,
237 "health_thread[service_id=%d]: disabled (health_interval_ms=0)",
238 ctx->service.service_id);
239 return POTR_SUCCESS; /* ヘルスチェック無効 */
240 }
241
242 POTR_LOG(POTR_LOG_DEBUG,
243 "health_thread[service_id=%d]: starting (interval=%ums)",
244 ctx->service.service_id,
245 (unsigned)ctx->global.health_interval_ms);
246
247#ifdef _WIN32
248 InitializeCriticalSection(&ctx->health_mutex);
249 InitializeConditionVariable(&ctx->health_wakeup);
250#else
251 pthread_mutex_init(&ctx->health_mutex, NULL);
252 pthread_cond_init(&ctx->health_wakeup, NULL);
253#endif
254
255 ctx->health_running = 1;
256
257#ifdef _WIN32
258 ctx->health_thread = CreateThread(NULL, 0, health_thread_func, ctx, 0, NULL);
259 if (ctx->health_thread == NULL)
260 {
261 ctx->health_running = 0;
262 POTR_LOG(POTR_LOG_ERROR,
263 "health_thread[service_id=%d]: CreateThread failed",
264 ctx->service.service_id);
265 return POTR_ERROR;
266 }
267#else
268 if (pthread_create(&ctx->health_thread, NULL, health_thread_func, ctx) != 0)
269 {
270 ctx->health_running = 0;
271 POTR_LOG(POTR_LOG_ERROR,
272 "health_thread[service_id=%d]: pthread_create failed",
273 ctx->service.service_id);
274 return POTR_ERROR;
275 }
276#endif
277
278 return POTR_SUCCESS;
279}

288int potr_health_thread_stop(struct PotrContext_ *ctx)
289{
290 if (ctx == NULL)
291 {
292 return POTR_ERROR;
293 }
294
295 if (!ctx->health_running)
296 {
297 return POTR_SUCCESS;
298 }
299
300 ctx->health_running = 0;
301
302#ifdef _WIN32
303 if (ctx->health_thread != NULL)
304 {
305 /* 条件変数をシグナルしてスリープ中のスレッドを即時起床させる */
306 EnterCriticalSection(&ctx->health_mutex);
307 WakeConditionVariable(&ctx->health_wakeup);
308 LeaveCriticalSection(&ctx->health_mutex);
309
310 WaitForSingleObject(ctx->health_thread, INFINITE);
311 CloseHandle(ctx->health_thread);
312 ctx->health_thread = NULL;
313 }
314 DeleteCriticalSection(&ctx->health_mutex);
315 /* Windows の CONDITION_VARIABLE は破棄不要 */
316#else
317 /* 条件変数をシグナルしてスリープ中のスレッドを即時起床させる */
318 pthread_mutex_lock(&ctx->health_mutex);
319 pthread_cond_signal(&ctx->health_wakeup);
320 pthread_mutex_unlock(&ctx->health_mutex);
321
322 pthread_join(ctx->health_thread, NULL);
323
324 pthread_cond_destroy(&ctx->health_wakeup);
325 pthread_mutex_destroy(&ctx->health_mutex);
326#endif
327
328 return POTR_SUCCESS;
329}
