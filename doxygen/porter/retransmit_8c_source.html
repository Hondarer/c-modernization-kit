114int retransmit_check(PotrRetransmit *rt, uint32_t now_ms, uint32_t *seq_num_out)
115{
116 uint32_t i;
117
118 if (rt == NULL || seq_num_out == NULL)
119 {
120 return POTR_ERROR;
121 }
122
123 for (i = 0; i < POTR_MAX_WINDOW_SIZE; i++)
124 {
125 if (!rt->entries[i].active)
126 {
127 continue;
128 }
129
130 if ((now_ms - rt->entries[i].send_time_ms) >= rt->timeout_ms)
131 {
132 if (rt->entries[i].retry_count >= rt->max_retry)
133 {
134 /* 最大再送回数超過 → 管理解除 */
135 rt->entries[i].active = 0;
136 continue;
137 }
138
139 rt->entries[i].retry_count++;
140 rt->entries[i].send_time_ms = now_ms;
141 *seq_num_out = rt->entries[i].seq_num;
142 return POTR_SUCCESS;
143 }
144 }
145
146 return POTR_ERROR;
147}
