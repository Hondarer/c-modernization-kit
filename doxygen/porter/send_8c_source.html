61static int parse_log_level(const char *str, PotrLogLevel *out)
62{
63 static const struct { const char *name; PotrLogLevel level; uint32_t _pad; } tbl[] = {
64 { "TRACE", POTR_LOG_TRACE, 0U },
65 { "DEBUG", POTR_LOG_DEBUG, 0U },
66 { "INFO", POTR_LOG_INFO, 0U },
67 { "WARN", POTR_LOG_WARN, 0U },
68 { "ERROR", POTR_LOG_ERROR, 0U },
69 { "FATAL", POTR_LOG_FATAL, 0U },
70 };
71 char upper[16];
72 size_t i;
73 size_t j;
74
75 for (j = 0; j < sizeof(upper) - 1U && str[j] != '\0'; j++)
76 {
77 upper[j] = (str[j] >= 'a' && str[j] <= 'z')
78 ? (char)(str[j] - ('a' - 'A'))
79 : str[j];
80 }
81 upper[j] = '\0';
82
83 for (i = 0; i < sizeof(tbl) / sizeof(tbl[0]); i++)
84 {
85 if (strcmp(upper, tbl[i].name) == 0)
86 {
87 *out = tbl[i].level;
88 return 1;
89 }
90 }
91 return 0;
92}
