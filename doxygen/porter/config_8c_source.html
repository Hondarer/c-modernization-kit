84/* 文字列の先頭・末尾の空白を除去して buf に格納する */
85static void trim(const char *src, char *buf, size_t buf_size)
86{
87 const char *start;
88 const char *end;
89 size_t len;
90
91 if (src == NULL || buf == NULL || buf_size == 0)
92 {
93 return;
94 }
95
96 start = src;
97 while (*start != '\0' && isspace((unsigned char)*start))
98 {
99 start++;
100 }
101
102 end = src + strlen(src) - 1;
103 while (end >= start && isspace((unsigned char)*end))
104 {
105 end--;
106 }
107
108 len = (size_t)(end - start + 1);
109 if (len >= buf_size)
110 {
111 len = buf_size - 1;
112 }
113
114 memcpy(buf, start, len);
115 buf[len] = '\0';
116}

118/* "key = value" 行を解析して key_out, val_out に格納する。成功時は 1 を返す。 */
119static int parse_kv(const char *line, char *key_out, size_t key_size,
120 char *val_out, size_t val_size)
121{
122 const char *eq;
123 char key_raw[CONFIG_KEY_MAX];
124 size_t key_len;
125
126 eq = strchr(line, '=');
127 if (eq == NULL)
128 {
129 return 0;
130 }
131
132 key_len = (size_t)(eq - line);
133 if (key_len >= CONFIG_KEY_MAX)
134 {
135 return 0;
136 }
137 memcpy(key_raw, line, key_len);
138 key_raw[key_len] = '\0';
139
140 trim(key_raw, key_out, key_size);
141 trim(eq + 1, val_out, val_size);
142
143 return 1;
144}

157int config_load_global(const char *config_path, PotrGlobalConfig *global)
158{
159 FILE *fp;
160 char line[CONFIG_LINE_MAX];
161 char section[CONFIG_SECTION_MAX];
162 char key[CONFIG_KEY_MAX];
163 char val[CONFIG_VAL_MAX];
164 int in_global;
165
166 if (config_path == NULL || global == NULL)
167 {
168 return POTR_ERROR;
169 }
170
171 /* デフォルト値で初期化 */
172 global->window_size = (uint16_t)POTR_DEFAULT_WINDOW_SIZE;
173 global->max_payload = (uint16_t)POTR_DEFAULT_MAX_PAYLOAD;
174 global->retransmit_timeout_ms = (uint32_t)POTR_DEFAULT_RETRANSMIT_TIMEOUT_MS;
175 global->retransmit_count = (uint8_t)POTR_DEFAULT_RETRANSMIT_COUNT;
176
177 fp = open_config_file_read(config_path);
178 if (fp == NULL)
179 {
180 return POTR_ERROR;
181 }
182
183 section[0] = '\0';
184 in_global = 0;
185
186 while (fgets(line, sizeof(line), fp) != NULL)
187 {
188 char trimmed[CONFIG_LINE_MAX];
189 trim(line, trimmed, sizeof(trimmed));
190
191 /* 空行・コメント行をスキップ */
192 if (trimmed[0] == '\0' || trimmed[0] == '#' || trimmed[0] == ';')
193 {
194 continue;
195 }
196
197 /* セクションヘッダー */
198 if (trimmed[0] == '[')
199 {
200 char *close = strchr(trimmed, ']');
201 if (close != NULL)
202 {
203 size_t sec_len = (size_t)(close - trimmed - 1);
204 if (sec_len >= CONFIG_SECTION_MAX)
205 {
206 sec_len = CONFIG_SECTION_MAX - 1;
207 }
208 memcpy(section, trimmed + 1, sec_len);
209 section[sec_len] = '\0';
210 in_global = (strcmp(section, "global") == 0) ? 1 : 0;
211 }
212 continue;
213 }
214
215 if (!in_global)
216 {
217 continue;
218 }
219
220 if (!parse_kv(trimmed, key, sizeof(key), val, sizeof(val)))
221 {
222 continue;
223 }
224
225 if (strcmp(key, "window_size") == 0)
226 {
227 global->window_size = (uint16_t)atoi(val);
228 }
229 else if (strcmp(key, "max_payload") == 0)
230 {
231 global->max_payload = (uint16_t)atoi(val);
232 }
233 else if (strcmp(key, "retransmit_timeout_ms") == 0)
234 {
235 global->retransmit_timeout_ms = (uint32_t)atoi(val);
236 }
237 else if (strcmp(key, "retransmit_count") == 0)
238 {
239 global->retransmit_count = (uint8_t)atoi(val);
240 }
241 }
242
243 fclose(fp);
244 return POTR_SUCCESS;
245}

248static void apply_service_kv(const char *key, const char *val,

264 }
265 }
266 else if (strcmp(key, "dst_port") == 0)
267 {
268 current->dst_port = (uint16_t)atoi(val);
269 }

271 {
272 current->src_port = (uint16_t)atoi(val);

275 {
276 copy_cstr_trunc(current->multicast_group,
277 sizeof(current->multicast_group),
278 val);
279 }
280 else if (strcmp(key, "ttl") == 0)

305int config_load_service(const char *config_path, int service_id,
306 PotrServiceDef *def)
307{
308 FILE *fp;
309 char line[CONFIG_LINE_MAX];
310 char key[CONFIG_KEY_MAX];
311 char val[CONFIG_VAL_MAX];
312 int in_target; /* 目的のサービスセクション内かどうか */
313 int found;
314
315 if (config_path == NULL || def == NULL)
316 {
317 return POTR_ERROR;
318 }
319
320 fp = open_config_file_read(config_path);
321 if (fp == NULL)
322 {
323 return POTR_ERROR;
324 }
325
326 in_target = 0;
327 found = 0;
328
329 while (fgets(line, sizeof(line), fp) != NULL)
330 {
331 char trimmed[CONFIG_LINE_MAX];
332 trim(line, trimmed, sizeof(trimmed));
333
334 if (trimmed[0] == '\0' || trimmed[0] == '#' || trimmed[0] == ';')
335 {
336 continue;
337 }
338
339 if (trimmed[0] == '[')
340 {
341 char section[CONFIG_SECTION_MAX];
342 char *close;
343 size_t sec_len;
344
345 /* 目的セクションを読み込み済みなら終了 */
346 if (in_target)
347 {
348 break;
349 }
350
351 close = strchr(trimmed, ']');
352 if (close != NULL)
353 {
354 sec_len = (size_t)(close - trimmed - 1);
355 if (sec_len >= CONFIG_SECTION_MAX)
356 {
357 sec_len = CONFIG_SECTION_MAX - 1;
358 }
359 memcpy(section, trimmed + 1, sec_len);
360 section[sec_len] = '\0';
361
362 /* [service.<id>] の <id> が service_id と一致するか判定 */
363 if (strncmp(section, "service.", 8) == 0 &&
364 atoi(section + 8) == service_id)
365 {
366 memset(def, 0, sizeof(*def));
367 def->ttl = (uint8_t)POTR_DEFAULT_TTL;
368 def->service_id = service_id;
369 in_target = 1;
370 found = 1;
371 }
372 }
373 continue;
374 }
375
376 if (!in_target)
377 {
378 continue;
379 }
380
381 if (!parse_kv(trimmed, key, sizeof(key), val, sizeof(val)))
382 {
383 continue;
384 }
385
386 apply_service_kv(key, val, def);
387 }
388
389 fclose(fp);
390 return found ? POTR_SUCCESS : POTR_ERROR;
391}
