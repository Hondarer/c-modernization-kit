118int packet_parse(PotrPacket *packet, const void *buf, size_t buf_len)
119{
120 const size_t header_size = sizeof(uint32_t) * 2 + sizeof(uint16_t) * 2;
121
122 if (packet == NULL || buf == NULL || buf_len < header_size)
123 {
124 return POTR_ERROR;
125 }
126
127 memcpy(packet, buf, buf_len < sizeof(*packet) ? buf_len : sizeof(*packet));
128
129 packet->seq_num = ntohl(packet->seq_num);
130 packet->ack_num = ntohl(packet->ack_num);
131 packet->flags = ntohs(packet->flags);
132 packet->payload_len = ntohs(packet->payload_len);
133
134 if (packet->payload_len > POTR_MAX_PAYLOAD)
135 {
136 return POTR_ERROR;
137 }
138
139 return POTR_SUCCESS;
140}
