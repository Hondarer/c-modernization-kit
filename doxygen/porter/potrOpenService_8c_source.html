37/* ソケットを作成して bind する。成功時は PotrSocket を返す。失敗時は POTR_INVALID_SOCKET。 */
38static PotrSocket open_socket_unicast(uint16_t port)
39{
40 PotrSocket sock;
41 struct sockaddr_in addr;
42 int reuse = 1;
43
44 sock = socket(AF_INET, SOCK_DGRAM, 0);
45 if (sock == POTR_INVALID_SOCKET)
46 {
47 return POTR_INVALID_SOCKET;
48 }
49
50#ifdef _WIN32
51 setsockopt(sock, SOL_SOCKET, SO_REUSEADDR,
52 (const char *)&reuse, sizeof(reuse));
53#else
54 setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, &reuse, sizeof(reuse));
55#endif
56
57 memset(&addr, 0, sizeof(addr));
58 addr.sin_family = AF_INET;
59 addr.sin_addr.s_addr = htonl(INADDR_ANY);
60 addr.sin_port = htons(port);
61
62 if (bind(sock, (struct sockaddr *)&addr, sizeof(addr)) < 0)
63 {
64#ifdef _WIN32
65 closesocket(sock);
66#else
67 close(sock);
68#endif
69 return POTR_INVALID_SOCKET;
70 }
71
72 return sock;
73}

186POTR_API int POTRAPI potrOpenService(const char *config_path,
187 int service_id,
188 PotrRecvCallback callback,
189 PotrHandle *handle)
190{
191 struct PotrContext_ *ctx;
192 PotrSocket sock;
193
194 if (config_path == NULL || handle == NULL)
195 {
196 return POTR_ERROR;
197 }
198
199#ifdef _WIN32
200 {
201 WSADATA wsa;
202 if (WSAStartup(MAKEWORD(2, 2), &wsa) != 0)
203 {
204 return POTR_ERROR;
205 }
206 }
207#endif
208
209 ctx = (struct PotrContext_ *)malloc(sizeof(struct PotrContext_));
210 if (ctx == NULL)
211 {
212 return POTR_ERROR;
213 }
214 memset(ctx, 0, sizeof(*ctx));
215 ctx->sock = POTR_INVALID_SOCKET;
216
217 /* 設定ファイルを読み込む */
218 if (config_load_global(config_path, &ctx->global) != POTR_SUCCESS)
219 {
220 free(ctx);
221 return POTR_ERROR;
222 }
223
224 if (config_load_service(config_path, service_id, &ctx->service) != POTR_SUCCESS)
225 {
226 free(ctx);
227 return POTR_ERROR;
228 }
229
230 /* 通信種別に応じてソケットを作成 */
231 switch (ctx->service.type)
232 {
233 case POTR_TYPE_UNICAST:
234 sock = open_socket_unicast(ctx->service.dst_port);
235 break;
236
237 case POTR_TYPE_MULTICAST:
238 sock = open_socket_multicast(&ctx->service);
239 break;
240
241 case POTR_TYPE_BROADCAST:
242 sock = open_socket_broadcast(ctx->service.src_port);
243 break;
244
245 default:
246 free(ctx);
247 return POTR_ERROR;
248 }
249
250 if (sock == POTR_INVALID_SOCKET)
251 {
252 free(ctx);
253 return POTR_ERROR;
254 }
255
256 ctx->sock = sock;
257 ctx->callback = callback;
258
259 /* ウィンドウ・再送制御を初期化 */
260 window_init(&ctx->send_window, 0, ctx->global.window_size);
261 window_init(&ctx->recv_window, 0, ctx->global.window_size);
262 retransmit_init(&ctx->retransmit,
263 ctx->global.retransmit_timeout_ms,
264 ctx->global.retransmit_count);
265
266 /* 受信コールバックが指定された場合は受信スレッドを起動 */
267 if (callback != NULL)
268 {
269 if (comm_recv_thread_start(ctx) != POTR_SUCCESS)
270 {
271#ifdef _WIN32
272 closesocket(ctx->sock);
273#else
274 close(ctx->sock);
275#endif
276 free(ctx);
277 return POTR_ERROR;
278 }
279 }
280
281 *handle = ctx;
282 return POTR_SUCCESS;
283}
