43/* ソケットを作成して bind する。成功時は PotrSocket を返す。失敗時は POTR_INVALID_SOCKET。
44 bind_addr: bind する IPv4 アドレス。port: bind するポート番号 (0 = OS 自動選定)。 */
45static PotrSocket open_socket_unicast(struct in_addr bind_addr, uint16_t port)
46{
47 PotrSocket sock;
48 struct sockaddr_in addr;
49 int reuse = 1;
50
51 sock = socket(AF_INET, SOCK_DGRAM, 0);
52 if (sock == POTR_INVALID_SOCKET)
53 {
54 return POTR_INVALID_SOCKET;
55 }
56
57#ifdef _WIN32
58 setsockopt(sock, SOL_SOCKET, SO_REUSEADDR,
59 (const char *)&reuse, sizeof(reuse));
60#else
61 setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, &reuse, sizeof(reuse));
62#endif
63
64 memset(&addr, 0, sizeof(addr));
65 addr.sin_family = AF_INET;
66 addr.sin_addr = bind_addr;
67 addr.sin_port = htons(port);
68
69 if (bind(sock, (struct sockaddr *)&addr, sizeof(addr)) < 0)
70 {
71#ifdef _WIN32
72 closesocket(sock);
73#else
74 close(sock);
75#endif
76 return POTR_INVALID_SOCKET;
77 }
78
79 return sock;
80}

82/* セッション識別子と開始時刻を生成してコンテキストに格納する。 */
83static void generate_session(struct PotrContext_ *ctx)
84{
85#ifdef _WIN32
86 FILETIME ft;
87 ULARGE_INTEGER uli;
88
89 srand((unsigned)(GetTickCount() ^ GetCurrentProcessId()));
90 ctx->session_id = (uint32_t)rand();
91
92 GetSystemTimeAsFileTime(&ft);
93 uli.LowPart = ft.dwLowDateTime;
94 uli.HighPart = ft.dwHighDateTime;
95 /* FILETIME: 100ns 単位、1601-01-01 起点。Unix エポックとの差: 11644473600 秒 */
96 ctx->session_tv_sec = (int64_t)(uli.QuadPart / 10000000ULL) - 11644473600LL;
97 ctx->session_tv_nsec = (int32_t)((uli.QuadPart % 10000000ULL) * 100ULL);
98#else
99 struct timespec ts;
100
101 srand((unsigned)((unsigned long)time(NULL) ^ (unsigned long)getpid()));
102 ctx->session_id = (uint32_t)rand();
103
104 clock_gettime(CLOCK_REALTIME, &ts);
105 ctx->session_tv_sec = (int64_t)ts.tv_sec;
106 ctx->session_tv_nsec = (int32_t)ts.tv_nsec;
107#endif
108}

110/* マルチキャストソケットを作成して bind・グループ参加する。
111 src_if: 使用するローカルインターフェース (INADDR_ANY = OS 自動選択)。
112 is_receiver: 1 = 受信者、0 = 送信者。 */
113static PotrSocket open_socket_multicast(const PotrServiceDef *def,
114 struct in_addr src_if,
115 int is_receiver)
116{
117 PotrSocket sock;
118 struct sockaddr_in addr;
119 struct ip_mreq mreq;
120 int reuse = 1;
121 /* 受信者: dst_port で bind する。送信者: src_port で bind する (送信元ポート)。 */
122 uint16_t bind_port;
123 if (is_receiver)
124 {
125 bind_port = def->dst_port;
126 }
127 else
128 {
129 bind_port = def->src_port;
130 }
131
132 sock = socket(AF_INET, SOCK_DGRAM, 0);
133 if (sock == POTR_INVALID_SOCKET)
134 {
135 return POTR_INVALID_SOCKET;
136 }
137
138#ifdef _WIN32
139 setsockopt(sock, SOL_SOCKET, SO_REUSEADDR,
140 (const char *)&reuse, sizeof(reuse));
141#else
142 setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, &reuse, sizeof(reuse));
143#endif
144
145 memset(&addr, 0, sizeof(addr));
146 addr.sin_family = AF_INET;
147 addr.sin_addr.s_addr = htonl(INADDR_ANY);
148 addr.sin_port = htons(bind_port);
149
150 if (bind(sock, (struct sockaddr *)&addr, sizeof(addr)) < 0)
151 {
152#ifdef _WIN32
153 closesocket(sock);
154#else
155 close(sock);
156#endif
157 return POTR_INVALID_SOCKET;
158 }
159
160 /* マルチキャストグループへ参加 (送受信ともに参加する) */
161 memset(&mreq, 0, sizeof(mreq));
162 if (parse_ipv4_addr(def->multicast_group, &mreq.imr_multiaddr) != POTR_SUCCESS)
163 {
164#ifdef _WIN32
165 closesocket(sock);
166#else
167 close(sock);
168#endif
169 return POTR_INVALID_SOCKET;
170 }
171 mreq.imr_interface = src_if;
172
173#ifdef _WIN32
174 if (setsockopt(sock, IPPROTO_IP, IP_ADD_MEMBERSHIP,
175 (const char *)&mreq, sizeof(mreq)) < 0)
176 {
177 closesocket(sock);
178 return POTR_INVALID_SOCKET;
179 }
180#else
181 if (setsockopt(sock, IPPROTO_IP, IP_ADD_MEMBERSHIP,
182 &mreq, sizeof(mreq)) < 0)
183 {
184 close(sock);
185 return POTR_INVALID_SOCKET;
186 }
187#endif
188
189 /* 送信者: マルチキャスト送信インターフェースを設定する */
190 if (!is_receiver)
191 {
192#ifdef _WIN32
193 setsockopt(sock, IPPROTO_IP, IP_MULTICAST_IF,
194 (const char *)&src_if, sizeof(src_if));
195#else
196 setsockopt(sock, IPPROTO_IP, IP_MULTICAST_IF,
197 &src_if, sizeof(src_if));
198#endif
199 }
200
201 return sock;
202}

204/* ブロードキャストソケットを作成して bind する。
205 src_port: 送信者の送信元 bind ポート (0 = OS 自動選定)。
206 dst_port: 受信者の listen ポート / 送信者の送信先ポート (省略不可)。
207 src_if: 送信者が使用するローカルインターフェース (INADDR_ANY = OS 自動選択)。
208 is_receiver: 1 = 受信者 (INADDR_ANY で bind)、0 = 送信者 (src_if で bind)。 */
209static PotrSocket open_socket_broadcast(uint16_t src_port,
210 uint16_t dst_port,
211 struct in_addr src_if,
212 int is_receiver)
213{
214 PotrSocket sock;
215 struct sockaddr_in addr;
216 int reuse = 1;
217 int bcast = 1;
218 /* 受信者: dst_port で bind する。送信者: src_port で bind する (送信元ポート)。 */
219 uint16_t bind_port;
220 if (is_receiver)
221 {
222 bind_port = dst_port;
223 }
224 else
225 {
226 bind_port = src_port;
227 }
228
229 sock = socket(AF_INET, SOCK_DGRAM, 0);
230 if (sock == POTR_INVALID_SOCKET)
231 {
232 return POTR_INVALID_SOCKET;
233 }
234
235#ifdef _WIN32
236 setsockopt(sock, SOL_SOCKET, SO_REUSEADDR,
237 (const char *)&reuse, sizeof(reuse));
238 setsockopt(sock, SOL_SOCKET, SO_BROADCAST,
239 (const char *)&bcast, sizeof(bcast));
240#else
241 setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, &reuse, sizeof(reuse));
242 setsockopt(sock, SOL_SOCKET, SO_BROADCAST, &bcast, sizeof(bcast));
243#endif
244
245 memset(&addr, 0, sizeof(addr));
246 addr.sin_family = AF_INET;
247 /* 送信者: src_addr で bind してインターフェースを選択する。受信者: INADDR_ANY で bind する。 */
248 if (!is_receiver)
249 {
250 addr.sin_addr.s_addr = src_if.s_addr;
251 }
252 else
253 {
254 addr.sin_addr.s_addr = htonl(INADDR_ANY);
255 }
256 addr.sin_port = htons(bind_port);
257
258 if (bind(sock, (struct sockaddr *)&addr, sizeof(addr)) < 0)
259 {
260#ifdef _WIN32
261 closesocket(sock);
262#else
263 close(sock);
264#endif
265 return POTR_INVALID_SOCKET;
266 }
267
268 return sock;
269}

271/* 生成済みソケットをすべてクローズする */
272static void cleanup_sockets(struct PotrContext_ *ctx)
273{
274 int i;
275 for (i = 0; i < (int)POTR_MAX_PATH; i++)
276 {
277 if (ctx->sock[i] != POTR_INVALID_SOCKET)
278 {
279#ifdef _WIN32
280 closesocket(ctx->sock[i]);
281#else
282 close(ctx->sock[i]);
283#endif
284 ctx->sock[i] = POTR_INVALID_SOCKET;
285 }
286 }
287}

289/* コンテキストが保持するすべてのリソースを解放して ctx 本体を free する。
290 memset(ctx, 0, ...) 後であれば、未初期化ポインタ (NULL) に対しても安全に呼び出せる。 */
291static void ctx_cleanup(struct PotrContext_ *ctx)
292{
293 window_destroy(&ctx->send_window);
294 window_destroy(&ctx->recv_window);
295 free(ctx->frag_buf);
296 free(ctx->compress_buf);
297 free(ctx->crypto_buf);
298 free(ctx->recv_buf);
299 free(ctx->send_wire_buf);
300 cleanup_sockets(ctx);
301 free(ctx);
302}

305POTR_API int POTRAPI potrOpenService(const char *config_path,
306 int service_id,
307 PotrRole role,
308 PotrRecvCallback callback,
309 PotrHandle *handle)
310{
311 struct PotrContext_ *ctx;
312
313 const char *config_path_log;
314 if (config_path != NULL)
315 {
316 config_path_log = config_path;
317 }
318 else
319 {
320 config_path_log = "(null)";
321 }
322 POTR_LOG(POTR_LOG_DEBUG,
323 "potrOpenService: service_id=%d role=%d config=%s",
324 service_id, (int)role,
325 config_path_log);
326
327 if (config_path == NULL || handle == NULL)
328 {
329 POTR_LOG(POTR_LOG_ERROR,
330 "potrOpenService: invalid argument (config_path=%p handle=%p)",
331 (const void *)config_path, (const void *)handle);
332 return POTR_ERROR;
333 }
334
335 /* role と callback の整合性チェック */
336 if (role == POTR_ROLE_RECEIVER && callback == NULL)
337 {
338 POTR_LOG(POTR_LOG_ERROR,
339 "potrOpenService: service_id=%d RECEIVER role requires callback",
340 service_id);
341 return POTR_ERROR;
342 }
343 if (role == POTR_ROLE_SENDER && callback != NULL)
344 {
345 POTR_LOG(POTR_LOG_ERROR,
346 "potrOpenService: service_id=%d SENDER role must not have callback",
347 service_id);
348 return POTR_ERROR;
349 }
350 if (role != POTR_ROLE_SENDER && role != POTR_ROLE_RECEIVER)
351 {
352 POTR_LOG(POTR_LOG_ERROR,
353 "potrOpenService: service_id=%d unknown role=%d",
354 service_id, (int)role);
355 return POTR_ERROR;
356 }
357
358#ifdef _WIN32
359 {
360 WSADATA wsa;
361 if (WSAStartup(MAKEWORD(2, 2), &wsa) != 0)
362 {
363 return POTR_ERROR;
364 }
365 }
366#endif
367
368 ctx = (struct PotrContext_ *)malloc(sizeof(struct PotrContext_));
369 if (ctx == NULL)
370 {
371 return POTR_ERROR;
372 }
373 memset(ctx, 0, sizeof(*ctx));
374
375 /* 全ソケットを INVALID で初期化 */
376 {
377 int i;
378 for (i = 0; i < (int)POTR_MAX_PATH; i++)
379 {
380 ctx->sock[i] = POTR_INVALID_SOCKET;
381 }
382 }
383
384 /* 設定ファイルを読み込む */
385 if (config_load_global(config_path, &ctx->global) != POTR_SUCCESS)
386 {
387 POTR_LOG(POTR_LOG_ERROR,
388 "potrOpenService: service_id=%d failed to load global config from '%s'",
389 service_id, config_path);
390 free(ctx);
391 return POTR_ERROR;
392 }
393
394 if (config_load_service(config_path, service_id, &ctx->service) != POTR_SUCCESS)
395 {
396 POTR_LOG(POTR_LOG_ERROR,
397 "potrOpenService: service_id=%d not found in '%s'",
398 service_id, config_path);
399 free(ctx);
400 return POTR_ERROR;
401 }
402
403 /* 設定値バリデーション */
404 if (ctx->global.max_payload < 64U || ctx->global.max_payload > POTR_MAX_PAYLOAD)
405 {
406 POTR_LOG(POTR_LOG_ERROR,
407 "potrOpenService: service_id=%d invalid max_payload=%u (range: 64..%u)",
408 service_id, (unsigned)ctx->global.max_payload, (unsigned)POTR_MAX_PAYLOAD);
409 free(ctx);
410 return POTR_ERROR;
411 }
412 if (ctx->global.window_size < 2U || ctx->global.window_size > POTR_MAX_WINDOW_SIZE)
413 {
414 POTR_LOG(POTR_LOG_ERROR,
415 "potrOpenService: service_id=%d invalid window_size=%u (range: 2..%u)",
416 service_id, (unsigned)ctx->global.window_size, (unsigned)POTR_MAX_WINDOW_SIZE);
417 free(ctx);
418 return POTR_ERROR;
419 }
420 if (ctx->global.max_message_size < (uint32_t)ctx->global.max_payload)
421 {
422 POTR_LOG(POTR_LOG_ERROR,
423 "potrOpenService: service_id=%d max_message_size=%u must be >= max_payload=%u",
424 service_id, (unsigned)ctx->global.max_message_size,
425 (unsigned)ctx->global.max_payload);
426 free(ctx);
427 return POTR_ERROR;
428 }
429 if (ctx->global.send_queue_depth < 2U)
430 {
431 POTR_LOG(POTR_LOG_ERROR,
432 "potrOpenService: service_id=%d invalid send_queue_depth=%u (min: 2)",
433 service_id, (unsigned)ctx->global.send_queue_depth);
434 free(ctx);
435 return POTR_ERROR;
436 }
437
438 POTR_LOG(POTR_LOG_DEBUG,
439 "potrOpenService: service_id=%d type=%d window=%u max_payload=%u"
440 " max_message_size=%u send_queue_depth=%u"
441 " health_interval=%ums health_timeout=%ums",
442 service_id, (int)ctx->service.type,
443 (unsigned)ctx->global.window_size, (unsigned)ctx->global.max_payload,
444 (unsigned)ctx->global.max_message_size, (unsigned)ctx->global.send_queue_depth,
445 (unsigned)ctx->global.health_interval_ms,
446 (unsigned)ctx->global.health_timeout_ms);
447
448 /* 通信種別に応じてソケットを作成 */
449 switch (ctx->service.type)
450 {
451 case POTR_TYPE_UNICAST:
452 {
453 int i;
454
455 if (ctx->service.dst_port == 0)
456 {
457 free(ctx);
458 return POTR_ERROR;
459 }
460
461 for (i = 0; i < (int)POTR_MAX_PATH; i++)
462 {
463 struct in_addr bind_addr;
464 uint16_t bind_port;
465
466 if (ctx->service.src_addr[i][0] == '\0' ||
467 ctx->service.dst_addr[i][0] == '\0')
468 {
469 break;
470 }
471
472 if (resolve_ipv4_addr(ctx->service.src_addr[i],
473 &ctx->src_addr_resolved[i]) != POTR_SUCCESS ||
474 resolve_ipv4_addr(ctx->service.dst_addr[i],
475 &ctx->dst_addr_resolved[i]) != POTR_SUCCESS)
476 {
477 ctx_cleanup(ctx);
478 return POTR_ERROR;
479 }
480
481 if (role == POTR_ROLE_RECEIVER)
482 {
483 bind_addr = ctx->dst_addr_resolved[i];
484 bind_port = ctx->service.dst_port;
485 }
486 else
487 {
488 bind_addr = ctx->src_addr_resolved[i];
489 bind_port = ctx->service.src_port;
490 }
491
492 ctx->sock[i] = open_socket_unicast(bind_addr, bind_port);
493 if (ctx->sock[i] == POTR_INVALID_SOCKET)
494 {
495 ctx_cleanup(ctx);
496 return POTR_ERROR;
497 }
498
499 ctx->n_path++;
500 }
501
502 if (ctx->n_path == 0)
503 {
504 ctx_cleanup(ctx);
505 return POTR_ERROR;
506 }
507 break;
508 }
509
510 case POTR_TYPE_MULTICAST:
511 {
512 int i;
513
514 if (ctx->service.dst_port == 0 ||
515 ctx->service.multicast_group[0] == '\0')
516 {
517 ctx_cleanup(ctx);
518 return POTR_ERROR;
519 }
520
521 for (i = 0; i < (int)POTR_MAX_PATH; i++)
522 {
523 if (ctx->service.src_addr[i][0] == '\0') break;
524
525 if (resolve_ipv4_addr(ctx->service.src_addr[i],
526 &ctx->src_addr_resolved[i]) != POTR_SUCCESS)
527 {
528 ctx_cleanup(ctx);
529 return POTR_ERROR;
530 }
531
532 ctx->sock[i] = open_socket_multicast(&ctx->service,
533 ctx->src_addr_resolved[i],
534 role == POTR_ROLE_RECEIVER);
535 if (ctx->sock[i] == POTR_INVALID_SOCKET)
536 {
537 ctx_cleanup(ctx);
538 return POTR_ERROR;
539 }
540
541 ctx->n_path++;
542 }
543
544 if (ctx->n_path == 0)
545 {
546 ctx_cleanup(ctx);
547 return POTR_ERROR;
548 }
549 break;
550 }
551
552 case POTR_TYPE_BROADCAST:
553 {
554 int i;
555
556 if (ctx->service.dst_port == 0)
557 {
558 ctx_cleanup(ctx);
559 return POTR_ERROR;
560 }
561
562 /* broadcast_addr 省略時は限定ブロードキャスト (255.255.255.255) を使用する */
563 if (ctx->service.broadcast_addr[0] == '\0')
564 {
565 const char *dflt = "255.255.255.255";
566 size_t len = strlen(dflt);
567 memcpy(ctx->service.broadcast_addr, dflt, len + 1);
568 }
569
570 for (i = 0; i < (int)POTR_MAX_PATH; i++)
571 {
572 if (ctx->service.src_addr[i][0] == '\0') break;
573
574 if (resolve_ipv4_addr(ctx->service.src_addr[i],
575 &ctx->src_addr_resolved[i]) != POTR_SUCCESS)
576 {
577 ctx_cleanup(ctx);
578 return POTR_ERROR;
579 }
580
581 ctx->sock[i] = open_socket_broadcast(ctx->service.src_port,
582 ctx->service.dst_port,
583 ctx->src_addr_resolved[i],
584 role == POTR_ROLE_RECEIVER);
585 if (ctx->sock[i] == POTR_INVALID_SOCKET)
586 {
587 ctx_cleanup(ctx);
588 return POTR_ERROR;
589 }
590
591 ctx->n_path++;
592 }
593
594 if (ctx->n_path == 0)
595 {
596 ctx_cleanup(ctx);
597 return POTR_ERROR;
598 }
599 break;
600 }
601
602 default:
603 ctx_cleanup(ctx);
604 return POTR_ERROR;
605 }
606
607 ctx->callback = callback;
608 ctx->role = role;
609
610 /* 送信者: 送信先ソケットアドレスを設定 */
611 if (role == POTR_ROLE_SENDER)
612 {
613 int i;
614
615 switch (ctx->service.type)
616 {
617 case POTR_TYPE_UNICAST:
618 for (i = 0; i < ctx->n_path; i++)
619 {
620 memset(&ctx->dest_addr[i], 0, sizeof(ctx->dest_addr[i]));
621 ctx->dest_addr[i].sin_family = AF_INET;
622 ctx->dest_addr[i].sin_addr = ctx->dst_addr_resolved[i];
623 ctx->dest_addr[i].sin_port = htons(ctx->service.dst_port);
624 }
625 break;
626
627 case POTR_TYPE_MULTICAST:
628 {
629 struct in_addr mcast_ip;
630 if (parse_ipv4_addr(ctx->service.multicast_group, &mcast_ip) != POTR_SUCCESS)
631 {
632 ctx_cleanup(ctx);
633 return POTR_ERROR;
634 }
635 for (i = 0; i < ctx->n_path; i++)
636 {
637 memset(&ctx->dest_addr[i], 0, sizeof(ctx->dest_addr[i]));
638 ctx->dest_addr[i].sin_family = AF_INET;
639 ctx->dest_addr[i].sin_addr = mcast_ip;
640 ctx->dest_addr[i].sin_port = htons(ctx->service.dst_port);
641 }
642 break;
643 }
644
645 case POTR_TYPE_BROADCAST:
646 {
647 struct in_addr bcast_ip;
648 if (parse_ipv4_addr(ctx->service.broadcast_addr, &bcast_ip) != POTR_SUCCESS)
649 {
650 ctx_cleanup(ctx);
651 return POTR_ERROR;
652 }
653 for (i = 0; i < ctx->n_path; i++)
654 {
655 memset(&ctx->dest_addr[i], 0, sizeof(ctx->dest_addr[i]));
656 ctx->dest_addr[i].sin_family = AF_INET;
657 ctx->dest_addr[i].sin_addr = bcast_ip;
658 ctx->dest_addr[i].sin_port = htons(ctx->service.dst_port);
659 }
660 break;
661 }
662
663 default:
664 break;
665 }
666 }
667
668 /* セッション識別子を生成する */
669 generate_session(ctx);
670
671 /* 送受信ウィンドウを初期化 */
672 if (window_init(&ctx->send_window, 0,
673 ctx->global.window_size, ctx->global.max_payload) != POTR_SUCCESS)
674 {
675 ctx_cleanup(ctx);
676 return POTR_ERROR;
677 }
678 if (window_init(&ctx->recv_window, 0,
679 ctx->global.window_size, ctx->global.max_payload) != POTR_SUCCESS)
680 {
681 ctx_cleanup(ctx);
682 return POTR_ERROR;
683 }
684
685 /* 動的バッファを確保 */
686 ctx->frag_buf = (uint8_t *)malloc(ctx->global.max_message_size);
687 if (ctx->frag_buf == NULL)
688 {
689 ctx_cleanup(ctx);
690 return POTR_ERROR;
691 }
692
693 ctx->compress_buf_size = POTR_COMPRESS_HEADER_SIZE
694 + (size_t)ctx->global.max_message_size + 64U;
695 ctx->compress_buf = (uint8_t *)malloc(ctx->compress_buf_size);
696 if (ctx->compress_buf == NULL)
697 {
698 ctx_cleanup(ctx);
699 return POTR_ERROR;
700 }
701
702 ctx->recv_buf = (uint8_t *)malloc(PACKET_HEADER_SIZE + ctx->global.max_payload);
703 if (ctx->recv_buf == NULL)
704 {
705 ctx_cleanup(ctx);
706 return POTR_ERROR;
707 }
708
709 ctx->send_wire_buf = (uint8_t *)malloc(PACKET_HEADER_SIZE + ctx->global.max_payload);
710 if (ctx->send_wire_buf == NULL)
711 {
712 ctx_cleanup(ctx);
713 return POTR_ERROR;
714 }
715
716 ctx->crypto_buf_size = ctx->global.max_payload + POTR_CRYPTO_TAG_SIZE;
717 ctx->crypto_buf = (uint8_t *)malloc(ctx->crypto_buf_size);
718 if (ctx->crypto_buf == NULL)
719 {
720 ctx_cleanup(ctx);
721 return POTR_ERROR;
722 }
723
724 /* 受信者の場合は受信スレッドを起動 */
725 if (role == POTR_ROLE_RECEIVER)
726 {
727 if (comm_recv_thread_start(ctx) != POTR_SUCCESS)
728 {
729 ctx_cleanup(ctx);
730 return POTR_ERROR;
731 }
732 }
733
734 /* 送信者: 送信キュー・送信スレッド・ヘルスチェックスレッド・ACK/NACK 受信スレッドを起動 */
735 if (role == POTR_ROLE_SENDER)
736 {
737 if (potr_send_queue_init(&ctx->send_queue,
738 (size_t)ctx->global.send_queue_depth,
739 ctx->global.max_payload) != POTR_SUCCESS)
740 {
741 ctx_cleanup(ctx);
742 return POTR_ERROR;
743 }
744
745 if (potr_send_thread_start(ctx) != POTR_SUCCESS)
746 {
747 potr_send_queue_destroy(&ctx->send_queue);
748 ctx_cleanup(ctx);
749 return POTR_ERROR;
750 }
751
752 if (potr_health_thread_start(ctx) != POTR_SUCCESS)
753 {
754 potr_send_thread_stop(ctx);
755 potr_send_queue_destroy(&ctx->send_queue);
756 ctx_cleanup(ctx);
757 return POTR_ERROR;
758 }
759
760 if (comm_recv_thread_start(ctx) != POTR_SUCCESS)
761 {
762 potr_health_thread_stop(ctx);
763 potr_send_thread_stop(ctx);
764 potr_send_queue_destroy(&ctx->send_queue);
765 ctx_cleanup(ctx);
766 return POTR_ERROR;
767 }
768 }
769
770 *handle = ctx;
771 const char *role_str;
772 if (role == POTR_ROLE_SENDER)
773 {
774 role_str = "SENDER";
775 }
776 else
777 {
778 role_str = "RECEIVER";
779 }
780 POTR_LOG(POTR_LOG_INFO,
781 "potrOpenService: service_id=%d role=%s encrypt=%s opened successfully",
782 service_id,
783 role_str,
784 ctx->service.encrypt_enabled ? "ON" : "OFF");
785 return POTR_SUCCESS;
786}
