37/* NACK パケットを送信元へユニキャスト送信する */
38static void send_nack(struct PotrContext_ *ctx,
39 const struct sockaddr_in *sender_addr,
40 uint32_t nack_seq)
41{
42 PotrPacket nack_pkt;
43 size_t wire_len;
44
45 if (packet_build_nack(&nack_pkt, nack_seq) != POTR_SUCCESS)
46 {
47 return;
48 }
49
50 wire_len = sizeof(uint32_t) * 2 + sizeof(uint16_t) * 2;
51
52#ifdef _WIN32
53 sendto(ctx->sock, (const char *)&nack_pkt, (int)wire_len, 0,
54 (const struct sockaddr *)sender_addr, sizeof(*sender_addr));
55#else
56 sendto(ctx->sock, &nack_pkt, wire_len, 0,
57 (const struct sockaddr *)sender_addr, sizeof(*sender_addr));
58#endif
59}

61/* 受信データを解凍してコールバックに渡す */
62static void recv_deliver(struct PotrContext_ *ctx,
63 const uint8_t *payload,
64 size_t payload_len,
65 int compressed)
66{
67 if (compressed)
68 {
69 size_t dec_len = sizeof(ctx->compress_buf);
70
71 if (potr_decompress(ctx->compress_buf, &dec_len,
72 payload, payload_len) == 0)
73 {
74 ctx->callback(ctx->service.service_id,
75 ctx->compress_buf,
76 dec_len);
77 }
78 /* 解凍失敗時はパケットを破棄する */
79 }
80 else
81 {
82 ctx->callback(ctx->service.service_id, payload, payload_len);
83 }
84}

86/* 受信スレッド本体 */
87#ifdef _WIN32
88static DWORD WINAPI recv_thread_func(LPVOID arg)
89#else
90static void *recv_thread_func(void *arg)
91#endif
92{
93 struct PotrContext_ *ctx = (struct PotrContext_ *)arg;
94 uint8_t buf[RECV_BUF_SIZE];
95 PotrPacket pkt;
96 PotrPacket ordered_pkt;
97 struct sockaddr_in sender_addr;
98 uint32_t nack_num;
99
100#ifdef _WIN32
101 int sender_len = sizeof(sender_addr);
102#else
103 socklen_t sender_len = sizeof(sender_addr);
104#endif
105
106 while (ctx->running)
107 {
108 int recv_len;
109
110 memset(&sender_addr, 0, sizeof(sender_addr));
111
112#ifdef _WIN32
113 recv_len = recvfrom(ctx->sock, (char *)buf, (int)sizeof(buf), 0,
114 (struct sockaddr *)&sender_addr, &sender_len);
115 if (recv_len == SOCKET_ERROR)
116 {
117 if (!ctx->running)
118 {
119 break;
120 }
121 continue;
122 }
123#else
124 recv_len = (int)recvfrom(ctx->sock, buf, sizeof(buf), 0,
125 (struct sockaddr *)&sender_addr, &sender_len);
126 if (recv_len < 0)
127 {
128 if (!ctx->running)
129 {
130 break;
131 }
132 continue;
133 }
134#endif
135
136 if (packet_parse(&pkt, buf, (size_t)recv_len) != POTR_SUCCESS)
137 {
138 continue;
139 }
140
141 /* ACK パケット: 送信ウィンドウを前進させる */
142 if (pkt.flags & POTR_FLAG_ACK)
143 {
144 window_send_ack(&ctx->send_window, pkt.ack_num);
145 continue;
146 }
147
148 /* NACK パケット: 再送制御に通知（再送は potrSend 側で実施） */
149 if (pkt.flags & POTR_FLAG_NACK)
150 {
151 /* 再送管理エントリのタイムアウトを即時化して次の送信で再送させる */
152 (void)pkt.ack_num;
153 continue;
154 }
155
156 /* データパケット: 受信ウィンドウへ格納 */
157 if (!(pkt.flags & POTR_FLAG_DATA))
158 {
159 continue;
160 }
161
162 if (window_recv_push(&ctx->recv_window, &pkt) != POTR_SUCCESS)
163 {
164 /* ウィンドウ外 → 無視 */
165 continue;
166 }
167
168 /* 欠番チェック → NACK 送信 */
169 if (window_recv_needs_nack(&ctx->recv_window, &nack_num))
170 {
171 send_nack(ctx, &sender_addr, nack_num);
172 }
173
174 /* 順序整列済みパケットをコールバックへ渡す（フラグメント結合・解凍対応） */
175 while (window_recv_pop(&ctx->recv_window, &ordered_pkt) == POTR_SUCCESS)
176 {
177 if (ordered_pkt.flags & POTR_FLAG_MORE_FRAG)
178 {
179 /* 後続フラグメントあり: バッファに蓄積する */
180 if (ctx->frag_buf_len + ordered_pkt.payload_len
181 <= POTR_MAX_MESSAGE_SIZE)
182 {
183 /* 最初のフラグメントで圧縮フラグを記録する */
184 if (ctx->frag_buf_len == 0)
185 {
186 ctx->frag_compressed =
187 (ordered_pkt.flags & POTR_FLAG_COMPRESSED) ? 1 : 0;
188 }
189 memcpy(ctx->frag_buf + ctx->frag_buf_len,
190 ordered_pkt.payload,
191 ordered_pkt.payload_len);
192 ctx->frag_buf_len += ordered_pkt.payload_len;
193 }
194 else
195 {
196 /* バッファオーバーフロー: 受信途中のフラグメントを破棄 */
197 ctx->frag_buf_len = 0;
198 ctx->frag_compressed = 0;
199 }
200 }
201 else if (ctx->frag_buf_len > 0)
202 {
203 /* フラグメントの最終パケット: バッファに追記してコールバック */
204 if (ctx->frag_buf_len + ordered_pkt.payload_len
205 <= POTR_MAX_MESSAGE_SIZE)
206 {
207 memcpy(ctx->frag_buf + ctx->frag_buf_len,
208 ordered_pkt.payload,
209 ordered_pkt.payload_len);
210 ctx->frag_buf_len += ordered_pkt.payload_len;
211
212 if (ctx->callback != NULL)
213 {
214 recv_deliver(ctx,
215 ctx->frag_buf,
216 ctx->frag_buf_len,
217 ctx->frag_compressed);
218 }
219 }
220 ctx->frag_buf_len = 0;
221 ctx->frag_compressed = 0;
222 }
223 else
224 {
225 /* 非フラグメントパケット */
226 if (ctx->callback != NULL)
227 {
228 recv_deliver(ctx,
229 ordered_pkt.payload,
230 (size_t)ordered_pkt.payload_len,
231 (ordered_pkt.flags & POTR_FLAG_COMPRESSED) ? 1 : 0);
232 }
233 }
234 }
235 }
236
237#ifdef _WIN32
238 return 0;
239#else
240 return NULL;
241#endif
242}

251int comm_recv_thread_start(struct PotrContext_ *ctx)
252{
253 if (ctx == NULL)
254 {
255 return POTR_ERROR;
256 }
257
258 ctx->running = 1;
259
260#ifdef _WIN32
261 ctx->recv_thread = CreateThread(NULL, 0, recv_thread_func, ctx, 0, NULL);
262 if (ctx->recv_thread == NULL)
263 {
264 ctx->running = 0;
265 return POTR_ERROR;
266 }
267#else
268 if (pthread_create(&ctx->recv_thread, NULL, recv_thread_func, ctx) != 0)
269 {
270 ctx->running = 0;
271 return POTR_ERROR;
272 }
273#endif
274
275 return POTR_SUCCESS;
276}

285int comm_recv_thread_stop(struct PotrContext_ *ctx)
286{
287 if (ctx == NULL)
288 {
289 return POTR_ERROR;
290 }
291
292 ctx->running = 0;
293
294#ifdef _WIN32
295 if (ctx->recv_thread != NULL)
296 {
297 /* ソケットをクローズして recvfrom をアンブロック */
298 closesocket(ctx->sock);
299 WaitForSingleObject(ctx->recv_thread, 3000);
300 CloseHandle(ctx->recv_thread);
301 ctx->recv_thread = NULL;
302 }
303#else
304 pthread_join(ctx->recv_thread, NULL);
305#endif
306
307 return POTR_SUCCESS;
308}
