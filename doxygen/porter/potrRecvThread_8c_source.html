64/* セッションの採用判定を行い、必要であればコンテキストの相手セッション情報を更新する。
65 採用すべきセッションなら 1、破棄すべき旧セッションなら 0 を返す。 */
66static int check_and_update_session(struct PotrContext_ *ctx,
67 const PotrPacket *pkt)
68{
69 if (!ctx->peer_session_known)
70 {
71 /* 初回受信 (または FIN/タイムアウト後の再接続): セッション採用 + ウィンドウをリセット。
72 pkt->seq_num で初期化することで、送信者の現在位置に直接同期し
73 NACK/REJECT サイクルなしに再加入できる。
74 DATA 着信時: window_init(DATA.seq_num) → push → pop → 即時 CONNECTED
75 PING 着信時: window_init(PING.seq_num) → gap スキャン範囲がゼロ → NACK なし
76 FIN/タイムアウト後は送信者が同一セッションのまま任意の seq から
77 再開する可能性があるため pkt->seq_num を使用する。 */
78 ctx->peer_session_id = pkt->session_id;
79 ctx->peer_session_tv_sec = pkt->session_tv_sec;
80 ctx->peer_session_tv_nsec = pkt->session_tv_nsec;
81 ctx->peer_session_known = 1;
82 POTR_LOG(POTR_LOG_TRACE,
83 "recv[service_id=%d]: new session (first contact), new_id=%u seq=%u",
84 ctx->service.service_id,
85 pkt->session_id, (unsigned)pkt->seq_num);
86 window_init(&ctx->recv_window, pkt->seq_num,
87 ctx->global.window_size, ctx->global.max_payload);
88 return 1;
89 }
90
91 /* (session_tv_sec, session_tv_nsec, session_id) の辞書順で新旧を判定する。
92 - pkt > 既知セッション: 新セッション。return せずにフォールスルーし、
93 関数末尾の「新セッション採用」ブロックで採用処理を行う。
94 - pkt < 既知セッション: 旧セッション。即 return 0 で破棄する。
95 - pkt == 既知セッション: 同一セッション。即 return 1 で通常受信を継続する。
96 新セッションと判定された分岐は LOG のみで return しないため、
97 if-else チェーンを抜けた後に必ず末尾の採用ブロックに到達する。 */
98 if (pkt->session_tv_sec > ctx->peer_session_tv_sec)
99 {
100 /* 新セッション (tv_sec が大): フォールスルーして採用 */
101 POTR_LOG(POTR_LOG_TRACE,
102 "recv[service_id=%d]: new session (tv_sec %lld > %lld)"
103 ", old_id=%u new_id=%u",
104 ctx->service.service_id,
105 (long long)pkt->session_tv_sec, (long long)ctx->peer_session_tv_sec,
106 ctx->peer_session_id, pkt->session_id);
107 }
108 else if (pkt->session_tv_sec < ctx->peer_session_tv_sec)
109 {
110 return 0; /* 旧セッション (tv_sec が小): 破棄 */
111 }
112 else if (pkt->session_tv_nsec > ctx->peer_session_tv_nsec)
113 {
114 /* 新セッション (tv_sec 同一・tv_nsec が大): フォールスルーして採用 */
115 POTR_LOG(POTR_LOG_TRACE,
116 "recv[service_id=%d]: new session (tv_nsec %d > %d)"
117 ", old_id=%u new_id=%u",
118 ctx->service.service_id,
119 pkt->session_tv_nsec, ctx->peer_session_tv_nsec,
120 ctx->peer_session_id, pkt->session_id);
121 }
122 else if (pkt->session_tv_nsec < ctx->peer_session_tv_nsec)
123 {
124 return 0; /* 旧セッション (tv_sec 同一・tv_nsec が小): 破棄 */
125 }
126 else if (pkt->session_id > ctx->peer_session_id)
127 {
128 /* 新セッション (タイムスタンプ完全一致・session_id が大): フォールスルーして採用 */
129 POTR_LOG(POTR_LOG_TRACE,
130 "recv[service_id=%d]: new session (id tiebreak %u > %u)",
131 ctx->service.service_id,
132 pkt->session_id, ctx->peer_session_id);
133 }
134 else
135 {
136 /* ここに到達するのは tv_sec == tv_sec かつ tv_nsec == tv_nsec かつ
137 session_id <= peer_session_id の場合のみ。
138 新セッション分岐はこの else には入らない。 */
139 if (pkt->session_id == ctx->peer_session_id)
140 {
141 return 1; /* 同一セッション: 採用済みのため再初期化不要 */
142 }
143 else
144 {
145 return 0; /* 旧セッション (タイムスタンプ完全一致・session_id が小): 破棄 */
146 }
147 }
148
149 /* 新セッション採用: コンテキストを更新しウィンドウをリセットする。
150 最初に受信したパケットの seq_num で初期化することで、送信者が先行して
151 送信済みの seq に直接同期し、不要な NACK/REJECT サイクルを発生させない。
152 送信者を先に起動して受信者が後から参加した場合も同様に機能する。 */
153 ctx->peer_session_id = pkt->session_id;
154 ctx->peer_session_tv_sec = pkt->session_tv_sec;
155 ctx->peer_session_tv_nsec = pkt->session_tv_nsec;
156 window_init(&ctx->recv_window, pkt->seq_num,
157 ctx->global.window_size, ctx->global.max_payload);
158 return 1;
159}

220/* タイムアウト時に経過時間を確認し、必要なら peer_port クリアと DISCONNECTED イベントを発火する */
221static void check_health_timeout(struct PotrContext_ *ctx)
222{
223 int64_t now_sec;
224 int32_t now_nsec;
225 int i;
226
227 if (ctx->global.health_timeout_ms == 0) return;
228
229 get_monotonic(&now_sec, &now_nsec);
230
231 /* パスごとのタイムアウト: peer_port をクリア */
232 for (i = 0; i < ctx->n_path; i++)
233 {
234 int64_t elapsed_ms;
235
236 if (ctx->path_last_recv_sec[i] == 0) continue;
237
238 elapsed_ms = (now_sec - ctx->path_last_recv_sec[i]) * 1000LL
239 + (now_nsec - ctx->path_last_recv_nsec[i]) / 1000000L;
240
241 if (elapsed_ms >= (int64_t)ctx->global.health_timeout_ms)
242 {
243 ctx->peer_port[i] = 0;
244 ctx->path_last_recv_sec[i] = 0;
245 }
246 }
247
248 /* 全体の health_alive 判定 */
249 if (!ctx->health_alive || ctx->last_recv_tv_sec == 0) return;
250
251 {
252 int64_t elapsed_ms;
253 elapsed_ms = (now_sec - ctx->last_recv_tv_sec) * 1000LL
254 + (now_nsec - ctx->last_recv_tv_nsec) / 1000000L;
255
256 if (elapsed_ms >= (int64_t)ctx->global.health_timeout_ms)
257 {
258 ctx->health_alive = 0;
259 POTR_LOG(POTR_LOG_WARN,
260 "recv[service_id=%d]: DISCONNECTED (timeout %lldms >= %ums)",
261 ctx->service.service_id,
262 (long long)elapsed_ms,
263 (unsigned)ctx->global.health_timeout_ms);
264 if (ctx->callback != NULL)
265 {
266 ctx->callback(ctx->service.service_id,
267 POTR_EVENT_DISCONNECTED, NULL, 0);
268 }
269
270 /* FIN と同様にセッション状態をリセットして次の接続を受け入れ可能にする。
271 peer_session_known をクリアすることで、送信者が同一セッションのまま
272 復帰した場合でも window_init を経由して受信ウィンドウを初期化し、
273 前セッションの next_seq が gap スキャンに影響しないようにする。 */
274 ctx->peer_session_known = 0;
275 ctx->last_recv_tv_sec = 0;
276 window_init(&ctx->recv_window, 0,
277 ctx->global.window_size, ctx->global.max_payload);
278 }
279 }

282/* NACK パケットを全パスへ送信する */
283static void send_nack(struct PotrContext_ *ctx, uint32_t nack_seq)
284{
285 PotrPacket nack_pkt;
286 PotrPacketSessionHdr shdr;
287 size_t wire_len;
288 int i;
289
290 shdr.service_id = ctx->service.service_id;
291 shdr.session_id = ctx->session_id;
292 shdr.session_tv_sec = ctx->session_tv_sec;
293 shdr.session_tv_nsec = ctx->session_tv_nsec;
294
295 if (packet_build_nack(&nack_pkt, &shdr, nack_seq) != POTR_SUCCESS) return;
296
297 wire_len = packet_wire_size(&nack_pkt);
298
299 for (i = 0; i < ctx->n_path; i++)
300 {
301 struct sockaddr_in dest;
302 uint16_t port;
303
304 if (ctx->service.src_port != 0)
305 {
306 port = htons(ctx->service.src_port);
307 }
308 else
309 {
310 port = ctx->peer_port[i]; /* NBO */
311 }
312
313 if (port == 0) continue; /* ポート未観測のパスは送れない */
314
315 memset(&dest, 0, sizeof(dest));
316 dest.sin_family = AF_INET;
317 dest.sin_addr = ctx->src_addr_resolved[i];
318 dest.sin_port = port;
319
320#ifdef _WIN32
321 sendto(ctx->sock[i], (const char *)&nack_pkt, (int)wire_len, 0,
322 (const struct sockaddr *)&dest, sizeof(dest));
323#else
324 sendto(ctx->sock[i], &nack_pkt, wire_len, 0,
325 (const struct sockaddr *)&dest, sizeof(dest));
326#endif
327 }
328}

396/* ペイロードエレメント 1 件のフラグメント結合・解凍・コールバック処理。
397 window_recv_pop で取り出した外側パケットを packet_unpack_next で展開した
398 各ペイロードエレメントに対して呼び出す。 */
399static void deliver_payload_elem(struct PotrContext_ *ctx, const PotrPacket *elem)
400{
401 if (elem->flags & POTR_FLAG_MORE_FRAG)
402 {
403 /* 中間フラグメント: バッファに追記 */
404 if (ctx->frag_buf_len + elem->payload_len <= ctx->global.max_message_size)
405 {
406 if (ctx->frag_buf_len == 0)
407 {
408 if (elem->flags & POTR_FLAG_COMPRESSED)
409 {
410 ctx->frag_compressed = 1;
411 }
412 else
413 {
414 ctx->frag_compressed = 0;
415 }
416 }
417 memcpy(ctx->frag_buf + ctx->frag_buf_len,
418 elem->payload, elem->payload_len);
419 ctx->frag_buf_len += elem->payload_len;
420 }
421 else
422 {
423 ctx->frag_buf_len = 0;
424 ctx->frag_compressed = 0;
425 }
426 }
427 else if (ctx->frag_buf_len > 0)
428 {
429 /* 最終フラグメント: バッファに追記してコールバック */
430 if (ctx->frag_buf_len + elem->payload_len <= ctx->global.max_message_size)
431 {
432 memcpy(ctx->frag_buf + ctx->frag_buf_len,
433 elem->payload, elem->payload_len);
434 ctx->frag_buf_len += elem->payload_len;
435
436 if (ctx->callback != NULL)
437 {
438 recv_deliver(ctx, ctx->frag_buf,
439 ctx->frag_buf_len, ctx->frag_compressed);
440 }
441 }
442 ctx->frag_buf_len = 0;
443 ctx->frag_compressed = 0;
444 }
445 else
446 {
447 /* フラグメントなし: 直接コールバック */
448 if (ctx->callback != NULL)
449 {
450 int is_compressed;
451 if (elem->flags & POTR_FLAG_COMPRESSED)
452 {
453 is_compressed = 1;
454 }
455 else
456 {
457 is_compressed = 0;
458 }
459 recv_deliver(ctx,
460 elem->payload,
461 (size_t)elem->payload_len,
462 is_compressed);
463 }
464 }
465}

467/* recv_window から順序整列済みの外側パケットを取り出してペイロードエレメントを配信する。
468 REJECT 処理後と通常受信処理の両方から呼び出す。 */
469static void drain_recv_window(struct PotrContext_ *ctx)
470{
471 PotrPacket pop_pkt;
472
473 while (window_recv_pop(&ctx->recv_window, &pop_pkt) == POTR_SUCCESS)
474 {
475 notify_health_alive(ctx);
476
477 const char *pkt_type_str;
478 if (pop_pkt.flags & POTR_FLAG_PING)
479 {
480 pkt_type_str = "PING";
481 }
482 else
483 {
484 pkt_type_str = "DATA";
485 }
486 POTR_LOG(POTR_LOG_TRACE,
487 "recv[service_id=%d]: pop seq=%u %s",
488 ctx->service.service_id,
489 (unsigned)pop_pkt.seq_num,
490 pkt_type_str);
491
492 if (pop_pkt.flags & POTR_FLAG_PING)
493 {
494 continue; /* PING: 生存確認のみ、ペイロードエレメント展開不要 */
495 }
496
497 /* DATA: ペイロードエレメントを順に展開して配信 */
498 {
499 size_t offset = 0;
500 PotrPacket elem;
501
502 while (packet_unpack_next(&pop_pkt, &offset, &elem) == POTR_SUCCESS)
503 {
504 deliver_payload_elem(ctx, &elem);
505 }
506 }
507 }
508}

510/* 外側パケット (DATA / PING) を受信ウィンドウに投入し、順序整列済みパケットを配信する。
511 再送・順序整列の単位は外側パケットであり、NACK も外側パケットの seq_num を指定する。 */
512static void process_outer_pkt(struct PotrContext_ *ctx,
513 const PotrPacket *pkt)
514{
515 uint32_t nack_num;
516 uint32_t stretch;
517
518 if (window_recv_push(&ctx->recv_window, pkt) != POTR_SUCCESS)
519 {
520 /* 通番がウィンドウ範囲外のためドロップ (受信ウィンドウ満杯、または古い重複パケット)。
521 受信者は next_seq を待ち続けるが、ヘルスチェックや後続パケット到着時に NACK が送られる。 */
522 POTR_LOG(POTR_LOG_ERROR,
523 "recv[service_id=%d]: recv_window full (100%%), dropping seq=%u"
524 " (base_seq=%u window_size=%u)",
525 ctx->service.service_id, (unsigned)pkt->seq_num,
526 (unsigned)ctx->recv_window.base_seq,
527 (unsigned)ctx->recv_window.window_size);
528 return;
529 }
530
531 /* ウィンドウ利用率チェック: 今回のパケットが占める先頭からの距離で推定する。
532 stretch = pkt->seq_num - base_seq + 1 。push 成功後は [1, window_size] の範囲。 */
533 stretch = (uint32_t)(pkt->seq_num - ctx->recv_window.base_seq) + 1U;
534 if (stretch * 10U >= (uint32_t)ctx->recv_window.window_size * 8U)
535 {
536 POTR_LOG(POTR_LOG_WARN,
537 "recv[service_id=%d]: recv_window utilization high (%u/%u >= 80%%)"
538 " seq=%u base_seq=%u",
539 ctx->service.service_id,
540 (unsigned)stretch, (unsigned)ctx->recv_window.window_size,
541 (unsigned)pkt->seq_num, (unsigned)ctx->recv_window.base_seq);
542 }
543
544 if (window_recv_needs_nack(&ctx->recv_window, &nack_num))
545 {
546 POTR_LOG(POTR_LOG_DEBUG,
547 "recv[service_id=%d]: NACK seq=%u",
548 ctx->service.service_id, (unsigned)nack_num);
549 send_nack(ctx, nack_num);
550 }
551
552 drain_recv_window(ctx);
553
554 /* drain 後に next_seq が前進した結果、新たな欠番が先頭に現れる場合は NACK を送る。
555 例: seq=3,4 欠落・seq=5 着時、drain 前は NACK(3)、seq=3 再送着→ drain で pop 後
556 next_seq=4 が欠番になるが次のパケット到着まで NACK(4) が遅延するのを防ぐ。 */
557 if (window_recv_needs_nack(&ctx->recv_window, &nack_num))
558 {
559 POTR_LOG(POTR_LOG_DEBUG,
560 "recv[service_id=%d]: NACK seq=%u (post-drain)",
561 ctx->service.service_id, (unsigned)nack_num);
562 send_nack(ctx, nack_num);
563 }
564}

580 {
581 poll_ms = ctx->global.health_timeout_ms / 3U;
582 if (poll_ms < 100U) poll_ms = 100U;
583 }
584 else
585 {
586 poll_ms = 500U;
587 }
588
589 while (ctx->running)
590 {
591 fd_set readfds;
592 struct timeval tv;
593 int ret;
594 int maxfd = -1;
595 int i;
596
597 FD_ZERO(&readfds);
598 for (i = 0; i < ctx->n_path; i++)
599 {
600 if (ctx->sock[i] == POTR_INVALID_SOCKET) continue;
601#ifdef _WIN32
602 FD_SET(ctx->sock[i], &readfds);
603 /* Windows では SOCKET は UINT_PTR。maxfd の代わりに n_path を使う */
604#else
605 FD_SET(ctx->sock[i], &readfds);
606 if (ctx->sock[i] > maxfd) maxfd = ctx->sock[i];
607#endif
608 }
609
610#ifdef _WIN32
611 /* Windows: select の第1引数は無視されるが 0 でなく n_path を渡す */
612 maxfd = ctx->n_path;
613#endif
614
615 if (maxfd < 0) break;
616
617 tv.tv_sec = (long)(poll_ms / 1000U);
618 tv.tv_usec = (long)((poll_ms % 1000U) * 1000U);
619
620 ret = select(maxfd + 1, &readfds, NULL, NULL, &tv);
621
622 if (ret < 0)
623 {
624 if (!ctx->running) break;
625 continue;
626 }
627
628 if (ret == 0)
629 {
630 if (ctx->role == POTR_ROLE_RECEIVER)
631 check_health_timeout(ctx);
632 continue;
633 }
634
635 for (i = 0; i < ctx->n_path; i++)
636 {
637 int recv_len;
638#ifdef _WIN32
639 int sender_len;
640#else
641 socklen_t sender_len;
642#endif
643
644 if (ctx->sock[i] == POTR_INVALID_SOCKET) continue;
645 if (!FD_ISSET(ctx->sock[i], &readfds)) continue;
646
647 memset(&sender_addr, 0, sizeof(sender_addr));
648 sender_len = sizeof(sender_addr);
649
650#ifdef _WIN32
651 recv_len = recvfrom(ctx->sock[i], (char *)buf,
652 (int)(PACKET_HEADER_SIZE + ctx->global.max_payload),
653 0, (struct sockaddr *)&sender_addr, &sender_len);
654#else
655 recv_len = (int)recvfrom(ctx->sock[i], buf,
656 PACKET_HEADER_SIZE + ctx->global.max_payload,
657 0, (struct sockaddr *)&sender_addr,
658 &sender_len);
659#endif
660 if (recv_len <= 0)
661 {
662 if (!ctx->running) break; /* 正常終了: ソケットクローズによる割り込み */
663 POTR_LOG(POTR_LOG_TRACE,
664 "recv[service_id=%d]: recvfrom returned %d",
665 ctx->service.service_id, recv_len);
666 continue;
667 }
668
669 if (packet_parse(&pkt, buf, (size_t)recv_len) != POTR_SUCCESS)
670 {
671 POTR_LOG(POTR_LOG_TRACE,
672 "recv[service_id=%d]: packet parse failed (len=%d)",
673 ctx->service.service_id, recv_len);
674 continue;
675 }
676 if (pkt.service_id != ctx->service.service_id)
677 {
678 POTR_LOG(POTR_LOG_TRACE,
679 "recv[service_id=%d]: ignored packet for service_id=%d",
680 ctx->service.service_id, pkt.service_id);
681 continue;
682 }
683 if (!check_src_addr(ctx, &sender_addr)) continue;
684
685 /* 暗号化パケットを復号する
686 * - POTR_FLAG_DATA | POTR_FLAG_ENCRYPTED の組み合わせのみ対象
687 * - 復号後 pkt.payload / pkt.payload_len を書き換えて以降の処理を透過させる
688 * - 認証失敗 (タグ不一致) は即座に破棄する
689 */
690 if ((pkt.flags & (POTR_FLAG_DATA | POTR_FLAG_ENCRYPTED))
691 == (POTR_FLAG_DATA | POTR_FLAG_ENCRYPTED))
692 {
693 uint8_t nonce[POTR_CRYPTO_NONCE_SIZE];
694 size_t dec_len = ctx->crypto_buf_size;
695 /* pkt.session_id / pkt.seq_num は packet_parse 後はホストオーダー */
696 uint32_t sid_nbo = htonl(pkt.session_id);
697 uint32_t seq_nbo = htonl(pkt.seq_num);
698
699 memcpy(nonce, &sid_nbo, 4);
700 memcpy(nonce + 4, &seq_nbo, 4);
701 memset(nonce + 8, 0, 4);
702
703 /* AAD = 受信 raw バイト先頭 32B (NBO、送信側と同一) */
704 if (potr_decrypt(ctx->crypto_buf, &dec_len,
705 pkt.payload, pkt.payload_len,
706 ctx->service.encrypt_key,
707 nonce,
708 buf, PACKET_HEADER_SIZE) != 0)
709 {
710 POTR_LOG(POTR_LOG_TRACE,
711 "recv[service_id=%d]: decrypt failed (auth) seq=%u",
712 ctx->service.service_id, (unsigned)pkt.seq_num);
713 continue;
714 }
715
716 pkt.payload = ctx->crypto_buf;
717 pkt.payload_len = (uint16_t)dec_len;
718 pkt.flags = (uint16_t)(pkt.flags & ~POTR_FLAG_ENCRYPTED);
719 }
720
721 /* ── 送信者ロール: NACK のみ処理 ── */
722 if (ctx->role == POTR_ROLE_SENDER)
723 {
724 if (pkt.flags & POTR_FLAG_NACK)
725 {
726 int j;
727
728 /* 同一 ack_num の NACK が POTR_NACK_DEDUP_MS 以内に届いた場合は破棄 */
729 {
730 uint64_t now_ms = get_ms_mono();
731 int dedup_idx;
732 int is_dup = 0;
733
734 for (dedup_idx = 0;
735 dedup_idx < (int)POTR_NACK_DEDUP_SLOTS;
736 dedup_idx++)
737 {
738 const PotrNackDedupEntry *e =
739 &ctx->nack_dedup_buf[dedup_idx];
740 if (e->time_ms != 0
741 && e->ack_num == pkt.ack_num
742 && (now_ms - e->time_ms)
743 < (uint64_t)POTR_NACK_DEDUP_MS)
744 {
745 is_dup = 1;
746 break;
747 }
748 }
749
750 if (is_dup)
751 {
752 continue;
753 }
754
755 ctx->nack_dedup_buf[ctx->nack_dedup_next].ack_num =
756 pkt.ack_num;
757 ctx->nack_dedup_buf[ctx->nack_dedup_next].time_ms =
758 now_ms;
759 ctx->nack_dedup_next =
760 (uint8_t)((ctx->nack_dedup_next + 1U)
761 % POTR_NACK_DEDUP_SLOTS);
762 }
763
764 {
765 PotrPacket resend_pkt;
766 int get_result;
767 size_t wire_len = 0;
768
769 /* send_window へのアクセスを排他制御する (送信スレッド・ヘルスチェックスレッドと競合)。
770 ミューテックス保持中に recv_buf へ wire データを組み立て、
771 プールスロットが上書きされる前にコピーを完了させる。 */
772#ifdef _WIN32
773 EnterCriticalSection(&ctx->send_window_mutex);
774#else
775 pthread_mutex_lock(&ctx->send_window_mutex);
776#endif
777 get_result = window_send_get(&ctx->send_window,
778 pkt.ack_num,
779 &resend_pkt);
780
781 if (get_result == POTR_SUCCESS)
782 {
783 /* [NBO ヘッダー 32B][ペイロード] を recv_buf に組み立てる */
784 wire_len = packet_wire_size(&resend_pkt);
785 memcpy(ctx->recv_buf, &resend_pkt, PACKET_HEADER_SIZE);
786 memcpy(ctx->recv_buf + PACKET_HEADER_SIZE,
787 resend_pkt.payload,
788 wire_len - PACKET_HEADER_SIZE);
789 }
790
791#ifdef _WIN32
792 LeaveCriticalSection(&ctx->send_window_mutex);
793#else
794 pthread_mutex_unlock(&ctx->send_window_mutex);
795#endif
796
797 if (get_result == POTR_SUCCESS)
798 {
799 POTR_LOG(POTR_LOG_DEBUG,
800 "sender[service_id=%d]: NACK received seq=%u"
801 " -> retransmit",
802 ctx->service.service_id,
803 (unsigned)pkt.ack_num);
804 for (j = 0; j < ctx->n_path; j++)
805 {
806#ifdef _WIN32
807 sendto(ctx->sock[j], (const char *)ctx->recv_buf,
808 (int)wire_len, 0,
809 (const struct sockaddr *)&ctx->dest_addr[j],
810 sizeof(ctx->dest_addr[j]));
811#else
812 sendto(ctx->sock[j], ctx->recv_buf, wire_len, 0,
813 (const struct sockaddr *)&ctx->dest_addr[j],
814 sizeof(ctx->dest_addr[j]));
815#endif
816 }
817 }
818 else
819 {
820 POTR_LOG(POTR_LOG_WARN,
821 "sender[service_id=%d]: NACK seq=%u not in window"
822 " -> REJECT",
823 ctx->service.service_id,
824 (unsigned)pkt.ack_num);
825 send_reject(ctx, pkt.ack_num);
826 }
827 }
828 }
829 /* ACK・その他 (DATA/PING の反射など) は無視 */
830 continue;
831 }
832
833 /* ── 受信者ロール: FIN / REJECT / DATA / PING を処理 ── */
834
835 /* FIN: 送信者からの正常終了通知 → 即座に DISCONNECTED へ遷移 */
836 if (pkt.flags & POTR_FLAG_FIN)
837 {
838 if (!check_and_update_session(ctx, &pkt))
839 {
840 continue; /* 旧セッションの FIN → 無視 */
841 }
842
843 POTR_LOG(POTR_LOG_INFO,
844 "recv[service_id=%d]: FIN received -> DISCONNECTED",
845 ctx->service.service_id);
846
847 /* health_alive で重複発火を防止する (ヘルスチェック有無によらず接続済み状態のみ) */
848 if (ctx->health_alive)
849 {
850 ctx->health_alive = 0;
851 if (ctx->callback != NULL)
852 {
853 ctx->callback(ctx->service.service_id,
854 POTR_EVENT_DISCONNECTED, NULL, 0);
855 }
856 }
857
858 /* セッション状態をリセットして次の接続を受け入れ可能にする。
859 受信ウィンドウも初期化して前セッションの next_seq を残さない。 */
860 ctx->peer_session_known = 0;
861 ctx->last_recv_tv_sec = 0;
862 window_init(&ctx->recv_window, 0,
863 ctx->global.window_size, ctx->global.max_payload);
864 continue;
865 }
866
867 /* REJECT: 欠落外側パケットをスキップして後続パケットを配信する */
868 if (pkt.flags & POTR_FLAG_REJECT)
869 {
870 if (!check_and_update_session(ctx, &pkt))
871 {
872 continue;
873 }
874
875 /* 送信元から受信できている = 生存確認としてタイムアウトをリセットする */
876 update_path_recv(ctx, i, &sender_addr);
877
878 POTR_LOG(POTR_LOG_WARN,
879 "recv[service_id=%d]: REJECT received seq=%u"
880 " (packet unrecoverable)",
881 ctx->service.service_id, (unsigned)pkt.ack_num);
882
883 /* DISCONNECTED イベント発火 (接続済みのときのみ。連続 REJECT で重複しない) */
884 if (ctx->health_alive)
885 {
886 ctx->health_alive = 0;
887 if (ctx->callback != NULL)
888 {
889 ctx->callback(ctx->service.service_id,
890 POTR_EVENT_DISCONNECTED, NULL, 0);
891 }
892 }
893
894 /* 欠落外側パケットをスキップして recv_window を前進させる */
895 window_recv_skip(&ctx->recv_window, pkt.ack_num);
896
897 /* 後続パケットを配信 (ウィンドウに溜まっていれば最初の pop で CONNECTED を発火) */
898 drain_recv_window(ctx);
899 continue;
900 }
901
902 /* NACK / ACK は受信者では無視 */
903 if (!(pkt.flags & (POTR_FLAG_DATA | POTR_FLAG_PING)))
904 {
905 continue;
906 }
907
908 /* セッション照合 */
909 if (!check_and_update_session(ctx, &pkt))
910 {
911 continue;
912 }
913
914 /* 受信時刻とパスポートを更新 (健全性タイムアウト計算用) */
915 update_path_recv(ctx, i, &sender_addr);
916
917 {
918 const char *pkt_kind_str;
919 if (pkt.flags & POTR_FLAG_PING)
920 {
921 pkt_kind_str = "PING";
922 }
923 else
924 {
925 pkt_kind_str = "DATA";
926 }
927 POTR_LOG(POTR_LOG_TRACE,
928 "recv[service_id=%d]: %s seq=%u path=%d",
929 ctx->service.service_id,
930 pkt_kind_str,
931 (unsigned)pkt.seq_num, i);
932 }
933
934 if (pkt.flags & POTR_FLAG_PING)
935 {
936 /* PING はウィンドウ外: NACK・再送の対象外。
937 seq_num は送信側の next_seq (消費前) を示す。
938 [recv_window.next_seq, seq_num) の範囲を全スキャンして欠番を一括 NACK する。 */
939 uint32_t scan_seq;
940
941 notify_health_alive(ctx);
942
943 scan_seq = ctx->recv_window.next_seq;
944 while (scan_seq != pkt.seq_num
945 && seqnum_in_window(scan_seq, ctx->recv_window.base_seq,
946 ctx->recv_window.window_size))
947 {
948 uint16_t idx = (uint16_t)((scan_seq - ctx->recv_window.base_seq)
949 % ctx->recv_window.window_size);
950 if (!ctx->recv_window.valid[idx])
951 {
952 POTR_LOG(POTR_LOG_DEBUG,
953 "recv[service_id=%d]: NACK seq=%u (PING gap scan)",
954 ctx->service.service_id, (unsigned)scan_seq);
955 send_nack(ctx, scan_seq);
956 }
957 scan_seq++;
958 }
959 }
960 else
961 {
962 /* DATA: ウィンドウ経由で順序整列・再送管理する */
963 process_outer_pkt(ctx, &pkt);
964 }
965 }
966 }
967
968#ifdef _WIN32
969 return 0;
970#else
971 return NULL;
972#endif
973}
